98     public void testWhenUninitializedLazyInitWithDefaultConfiguration() throws IOException {
99        AdminAccess access = AdminAccess.standard();
100       // before rendered, holder is empty
101       assertEquals(access.config.getAdminUsername(), null);
102       assertEquals(access.config.getAdminPassword(), null);
103       assertEquals(access.config.getAdminPublicKey(), null);
104       assertEquals(access.config.getAdminPrivateKey(), null);
105       assertEquals(access.config.getLoginPassword(), null);
106       access.render(OsFamily.UNIX);
107       // DefaultConfiguration
108       try {
109          assertEquals(access.config.getAdminUsername(), System.getProperty("user.name"));
110          assertNotNull(access.config.getAdminPassword());
111          assertNotNull(access.config.getAdminPublicKey());
112          assertNotNull(access.config.getAdminPrivateKey());
113          assertNotNull(access.config.getLoginPassword());
114       } catch (AssertionError e) {
115          throw e;
116       } catch (Throwable e) {
117          // we are catching throwables here, in case the test runner doesn't
118          // have ssh keys setup
119       }

47  /**
48   * Creates a run script
49   * 
50   * @author Adrian Cole
51   */
52  public class CreateRunScript extends StatementList {
53     public final static String DELIMETER = "END_OF_JCLOUDS_SCRIPT";
54     final String instanceName;
55     final Iterable<String> exports;
56     final String pwd;
57  
58     public CreateRunScript(String instanceName, Iterable<String> exports, String pwd, Iterable<Statement> statements) {
59        super(statements);
60        this.instanceName = checkNotNull(instanceName, "instanceName");
61        this.exports = checkNotNull(exports, "exports");
62        this.pwd = checkNotNull(pwd, "pwd").replaceAll("[/\\\\]", "{fs}");
63     }
64  
65     public static class AddExport implements Statement {
66        final String export;
67        final String value;
68  
69        public AddExport(String export, String value) {
70           this.export = checkNotNull(CaseFormat.LOWER_CAMEL.to(CaseFormat.UPPER_UNDERSCORE, export), "export");
71           this.value = checkNotNull(value, "value");
72        }
73  
74        public static final Map<OsFamily, String> OS_TO_EXPORT_PATTERN = ImmutableMap.of(OsFamily.UNIX,
75              "export {export}='{value}'\n", OsFamily.WINDOWS, "set {export}={value}\r\n");
76  
77        @Override
78        public Iterable<String> functionDependencies(OsFamily family) {
79           return Collections.emptyList();
80        }
81  
82        @Override
83        public String render(OsFamily family) {
84           return Utils
85                 .replaceTokens(OS_TO_EXPORT_PATTERN.get(family), ImmutableMap.of("export", export, "value", value));
86        }
87     }
88  
89     @Override
90     public Iterable<String> functionDependencies(OsFamily family) {
91        return Collections.emptyList();
92     }
93  
94     @Override
95     public String render(OsFamily family) {
96        if (checkNotNull(family, "family") == OsFamily.WINDOWS)
97           throw new UnsupportedOperationException("windows not yet implemented");
98        List<Statement> statements = newArrayList();
99        final Map<String, String> tokenMap = ShellToken.tokenValueMap(family);
100       String runScript = Utils.replaceTokens(pwd + "{fs}" + instanceName + ".{sh}", tokenMap);
101       statements.add(interpret(String.format("{md} %s{lf}", pwd)));
102       StringBuilder builder = new StringBuilder();
103       builder.append("\n");
104       addUnixRunScriptHeader(runScript, builder);
105       builder.append("\n");
106       addUnixRunScript(runScript, builder);
107       builder.append("\n");
108       addUnixRunScriptFooter(runScript, builder);
109       builder.append("\n");
110       statements.add(interpret(builder.toString()));
111       statements.add(exec("chmod u+x " + runScript));
112       return new StatementList(statements).render(family);
113    }
114 
115    private void addUnixRunScriptFooter(String runScript, StringBuilder builder) {
116       builder.append("# add runscript footer\n");
117       Iterable<String> endScript = Splitter.on(ShellToken.LF.to(OsFamily.UNIX)).split(
118             ShellToken.END_SCRIPT.to(OsFamily.UNIX));
119       builder.append(appendFile(runScript, endScript, DELIMETER).render(OsFamily.UNIX));
120    }
121 
122    private void addUnixRunScript(String runScript, StringBuilder builder) {
123       builder.append("# add desired commands from the user\n");
124       Builder<String> userCommands = ImmutableList.builder();
125       userCommands.add("cd " + pwd);
126       for (Statement statement : statements) {
127          if (statement instanceof Call
128                || (statement instanceof StatementList && any(StatementList.class.cast(statement).delegate(),
129                      instanceOf(Call.class)))) {
130             statement = new ExitInsteadOfReturn(statement);
131          }
132          userCommands.addAll(Splitter.on('\n').split(statement.render(OsFamily.UNIX)));
133       }
134       builder.append(appendFile(runScript, userCommands.build(), DELIMETER).render(OsFamily.UNIX));
135    }
136 
137    private void addUnixRunScriptHeader(String runScript, StringBuilder builder) {
138       builder.append("# create runscript header\n");
139 
140       Builder<String> beginningOfFile = ImmutableList.builder();
141       beginningOfFile.addAll(Splitter.on(ShellToken.LF.to(OsFamily.UNIX)).split(
142             ShellToken.BEGIN_SCRIPT.to(OsFamily.UNIX)));
143       beginningOfFile.add(format("PROMPT_COMMAND='echo -ne \\\"\\033]0;%s\\007\\\"'", instanceName));
144       beginningOfFile.add(Utils.writeZeroPath(OsFamily.UNIX));
145       beginningOfFile.add(format("export INSTANCE_NAME='%s'", instanceName));
146       builder.append(createOrOverwriteFile(runScript, beginningOfFile.build(), DELIMETER).render(OsFamily.UNIX));
147 
148       // expanding variables here.
149       builder.append(AppendFile.builder().path(runScript).delimeter(DELIMETER).expandVariables(true)
150             .lines(Iterables.transform(exports, new Function<String, String>() {
151 
152                @Override
153                public String apply(String export) {
154                   String variableNameInUpper = CaseFormat.LOWER_CAMEL.to(CaseFormat.UPPER_UNDERSCORE, export);
155                   return new StringBuilder().append("export ").append(variableNameInUpper).append("='$")
156                         .append(variableNameInUpper).append("'").toString();
157                }
158             })).build().render(OsFamily.UNIX));
159 
160       Map<String, String> functionsToWrite = ScriptBuilder.resolveFunctionDependenciesForStatements(
161             ImmutableMap.<String, String> of("abort", Utils.writeFunctionFromResource("abort", OsFamily.UNIX)),
162             statements, OsFamily.UNIX);
163 
164       // if there are more functions than simply abort
165       if (functionsToWrite.size() > 1) {
166          StringBuilder functions = new StringBuilder();
167          ScriptBuilder.writeFunctions(functionsToWrite, OsFamily.UNIX, functions);
168          builder.append(appendFile(runScript, functions.toString(), DELIMETER).render(OsFamily.UNIX));
169       }
170    }

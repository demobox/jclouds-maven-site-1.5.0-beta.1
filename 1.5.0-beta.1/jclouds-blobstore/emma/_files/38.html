377      public boolean apply(StorageMetadata metadata) {

379            return metadata.getName().indexOf(delimiter) == -1;
380         // ensure we don't accidentally append twice

389         }

391      }

487   private Blob createUpdatedCopyOfBlobInContainer(String containerName, Blob in) {
488      checkNotNull(in, "blob");
489      checkNotNull(in.getPayload(), "blob.payload");
490      ByteArrayPayload payload = (in.getPayload() instanceof ByteArrayPayload) ? ByteArrayPayload.class.cast(in
491               .getPayload()) : null;
492      if (payload == null)
493         payload = (in.getPayload() instanceof DelegatingPayload) ? (DelegatingPayload.class.cast(in.getPayload())
494                  .getDelegate() instanceof ByteArrayPayload) ? ByteArrayPayload.class.cast(DelegatingPayload.class
495                  .cast(in.getPayload()).getDelegate()) : null : null;
496      try {
497         if (payload == null || !(payload instanceof ByteArrayPayload)) {
498            MutableContentMetadata oldMd = in.getPayload().getContentMetadata();
499            ByteArrayOutputStream out = new ByteArrayOutputStream();
500            in.getPayload().writeTo(out);
501            payload = (ByteArrayPayload) Payloads.calculateMD5(Payloads.newPayload(out.toByteArray()));
502            HttpUtils.copy(oldMd, payload.getContentMetadata());
503         } else {
504            if (payload.getContentMetadata().getContentMD5() == null)
505               Payloads.calculateMD5(in, crypto.md5());
506         }
507      } catch (IOException e) {
508         Throwables.propagate(e);
509      }
510      Blob blob = blobFactory.create(copy(in.getMetadata()));
511      blob.setPayload(payload);
512      blob.getMetadata().setContainer(containerName);
513      blob.getMetadata().setUri(
514               uriBuilders.get().scheme("mem").host(containerName).path(in.getMetadata().getName()).build());
515      blob.getMetadata().setLastModified(new Date());
516      String eTag = CryptoStreams.hex(payload.getContentMetadata().getContentMD5());
517      blob.getMetadata().setETag(eTag);
518      // Set HTTP headers to match metadata
519      blob.getAllHeaders().replaceValues(HttpHeaders.LAST_MODIFIED,
520               Collections.singleton(dateService.rfc822DateFormat(blob.getMetadata().getLastModified())));
521      blob.getAllHeaders().replaceValues(HttpHeaders.ETAG, Collections.singleton(eTag));
522      copyPayloadHeadersToBlob(payload, blob);
523      blob.getAllHeaders().putAll(Multimaps.forMap(blob.getMetadata().getUserMetadata()));
524      return blob;
525   }

542    * {@inheritDoc}
543    */
544   @Override
545   public ListenableFuture<Blob> getBlob(final String containerName, final String key, GetOptions options) {
546      logger.debug("Retrieving blob with key %s from container %s", key, containerName);
547      // If the container doesn't exist, an exception is thrown
548      if (!storageStrategy.containerExists(containerName)) {
549         logger.debug("Container %s does not exist", containerName);
550         return immediateFailedFuture(cnfe(containerName));
551      }
552      // If the blob doesn't exist, a null object is returned
553      if (!storageStrategy.blobExists(containerName, key)) {
554         logger.debug("Item %s does not exist in container %s", key, containerName);
555         return immediateFuture(null);
556      }
557 
558      Blob blob = loadBlob(containerName, key);
559 
560      if (options != null) {
561         if (options.getIfMatch() != null) {
562            if (!blob.getMetadata().getETag().equals(options.getIfMatch()))
563               return immediateFailedFuture(returnResponseException(412));
564         }
565         if (options.getIfNoneMatch() != null) {
566            if (blob.getMetadata().getETag().equals(options.getIfNoneMatch()))
567               return immediateFailedFuture(returnResponseException(304));
568         }
569         if (options.getIfModifiedSince() != null) {
570            Date modifiedSince = options.getIfModifiedSince();
571            if (blob.getMetadata().getLastModified().before(modifiedSince)) {
572               HttpResponse response = new HttpResponse(304, null, null);
573               return immediateFailedFuture(new HttpResponseException(String.format("%1$s is before %2$s", blob
574                     .getMetadata().getLastModified(), modifiedSince), null, response));
575            }
576 
577         }
578         if (options.getIfUnmodifiedSince() != null) {
579            Date unmodifiedSince = options.getIfUnmodifiedSince();
580            if (blob.getMetadata().getLastModified().after(unmodifiedSince)) {
581               HttpResponse response = new HttpResponse(412, null, null);
582               return immediateFailedFuture(new HttpResponseException(String.format("%1$s is after %2$s", blob
583                     .getMetadata().getLastModified(), unmodifiedSince), null, response));
584            }
585         }
586         blob = copyBlob(blob);
587 
588         if (options.getRanges() != null && options.getRanges().size() > 0) {
589            byte[] data;
590            try {
591               data = toByteArray(blob.getPayload().getInput());
592            } catch (IOException e) {
593               return immediateFailedFuture(new RuntimeException(e));
594            }
595            ByteArrayOutputStream out = new ByteArrayOutputStream();
596            for (String s : options.getRanges()) {
597               // HTTP uses a closed interval while Java array indexing uses a
598               // half-open interval.
599               int offset = 0;
600               int last = data.length - 1;
601               if (s.startsWith("-")) {
602                  offset = last - Integer.parseInt(s.substring(1)) + 1;
603               } else if (s.endsWith("-")) {
604                  offset = Integer.parseInt(s.substring(0, s.length() - 1));
605               } else if (s.contains("-")) {
606                  String[] firstLast = s.split("\\-");
607                  offset = Integer.parseInt(firstLast[0]);
608                  last = Integer.parseInt(firstLast[1]);
609               } else {
610                  return immediateFailedFuture(new IllegalArgumentException("illegal range: " + s));
611               }
612 
613               if (offset > last) {
614                  return immediateFailedFuture(new IllegalArgumentException("illegal range: " + s));
615               }
616               if (last + 1 > data.length) {
617                  last = data.length - 1;
618               }
619               out.write(data, offset, last - offset + 1);
620            }
621            ContentMetadata cmd = blob.getPayload().getContentMetadata();
622            byte[] byteArray = out.toByteArray();
623            blob.setPayload(byteArray);
624            HttpUtils.copy(cmd, blob.getPayload().getContentMetadata());
625            blob.getPayload().getContentMetadata().setContentLength(new Long(byteArray.length));
626         }
627      }
628      checkNotNull(blob.getPayload(), "payload " + blob);
629      return immediateFuture(blob);
630   }

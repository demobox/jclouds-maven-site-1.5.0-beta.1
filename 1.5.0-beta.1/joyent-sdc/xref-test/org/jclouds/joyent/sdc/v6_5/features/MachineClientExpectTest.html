39  @Test(groups = "unit", testName = "MachineClientExpectTest")
40  public class MachineClientExpectTest extends BaseSDCClientExpectTest {
41     HttpRequest listMachines = HttpRequest.builder().method("GET").endpoint(
42              URI.create("https://api.joyentcloud.com/my/machines")).headers(
43              ImmutableMultimap.<String, String> builder().put("X-Api-Version", "~6.5").put("Accept", "application/json")
44                       .put("Authorization", "Basic aWRlbnRpdHk6Y3JlZGVudGlhbA==").build()).build();
45  
46     public void testListMachinesWhenResponseIs2xx() {
47        HttpResponse listMachinesResponse = HttpResponse.builder().statusCode(200).payload(
48                 payloadFromResource("/machine_list.json")).build();
49  
50        SDCClient clientWhenMachinesExists = requestSendsResponse(listMachines, listMachinesResponse);
51  
52        assertEquals(clientWhenMachinesExists.getMachineClient().listMachines(), new ParseMachineListTest().expected());
53     }
54  
55     public void testListMachinesWhenResponseIs404() {
56        HttpResponse listMachinesResponse = HttpResponse.builder().statusCode(404).build();
57  
58        SDCClient listMachinesWhenNone = requestSendsResponse(listMachines, listMachinesResponse);
59  
60        assertEquals(listMachinesWhenNone.getMachineClient().listMachines(), ImmutableSet.of());
61     }
62  
63     public void testCreateMachineWhenResponseIs202() throws Exception {
64        HttpRequest createMachine = HttpRequest
65                 .builder()
66                 .method("POST")
67                 .endpoint(URI.create("https://api.joyentcloud.com/my/machines"))
68                 .headers(
69                          ImmutableMultimap.<String, String> builder().put("X-Api-Version", "~6.5").put("Accept",
70                                   "application/json").put("Authorization", "Basic aWRlbnRpdHk6Y3JlZGVudGlhbA==").build())
71                 .payload(
72                          payloadFromStringWithContentType(
73                                   "{\"name\":\"testJClouds\",\"package\":\"Small 1GB\",\"dataset\":\"sdc:sdc:centos-5.7:1.2.1\"}",
74                                   "application/json")).build();
75  
76        HttpResponse createMachineResponse = HttpResponse.builder().statusCode(202).message("HTTP/1.1 202 Accepted")
77                 .payload(payloadFromResourceWithContentType("/new_machine.json", "application/json; charset=UTF-8"))
78                 .build();
79  
80        SDCClient clientWithNewMachine = requestSendsResponse(createMachine, createMachineResponse);
81  
82        assertEquals(clientWithNewMachine.getMachineClient().createMachine("testJClouds", "Small 1GB",
83                 "sdc:sdc:centos-5.7:1.2.1").toString(), new ParseCreatedMachineTest().expected().toString());
84     }

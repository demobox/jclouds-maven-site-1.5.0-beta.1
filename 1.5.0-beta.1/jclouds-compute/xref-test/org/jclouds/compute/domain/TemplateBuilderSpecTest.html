624    private void assertTemplateBuilderEquivalence(TemplateBuilder a, TemplateBuilder b) {
625       // Labs hack: dig into the TemplateBuilder instances, verifying all fields
626       // are equal.
627       for (Field f : TemplateBuilder.class.getFields()) {
628          f.setAccessible(true);
629          try {
630             assertEquals(f.get(a), f.get(b), "Field " + f.getName() + " not equal");
631          } catch (IllegalArgumentException e) {
632             throw new AssertionError(e.getMessage());
633          } catch (IllegalAccessException e) {
634             throw new AssertionError(e.getMessage());
635          }
636       }

190    @Override
191    public Set<? extends NodeMetadata> createNodesInGroup(String group, int count, Template template)
192          throws RunNodesException {
193       checkNotNull(group, "group cannot be null");
194       checkNotNull(template.getLocation(), "location");
195       logger.debug(">> running %d node%s group(%s) location(%s) image(%s) hardwareProfile(%s) options(%s)", count,
196             count > 1 ? "s" : "", group, template.getLocation().getId(), template.getImage().getId(), template
197                   .getHardware().getId(), template.getOptions());
198       Set<NodeMetadata> goodNodes = newLinkedHashSet();
199       Map<NodeMetadata, Exception> badNodes = newLinkedHashMap();
200       Multimap<NodeMetadata, CustomizationResponse> customizationResponses = LinkedHashMultimap.create();
201 
202       if (template.getOptions().getRunScript() != null)
203          initAdminAccess.visit(template.getOptions().getRunScript());
204 
205       Map<?, Future<Void>> responses = runNodesAndAddToSetStrategy.execute(group, count, template, goodNodes, badNodes,
206             customizationResponses);
207       Map<?, Exception> executionExceptions = awaitCompletion(responses, executor, null, logger, "createNodesInGroup("
208             + group + ")");
209       Function<NodeMetadata, NodeMetadata> fn = persistNodeCredentials.always(template.getOptions().getRunScript());
210       badNodes = Maps2.transformKeys(badNodes, fn);
211       goodNodes = ImmutableSet.copyOf(Iterables.transform(goodNodes, fn));
212       if (executionExceptions.size() > 0 || badNodes.size() > 0) {
213          throw new RunNodesException(group, count, template, goodNodes, executionExceptions, badNodes);
214       }
215       return goodNodes;
216    }

245       Set<NodeMetadata> set = newLinkedHashSet(filter(transformParallel(nodesMatchingFilterAndNotTerminated(filter),

274    /**
275     * 
276     * @param id
277     * @return node that was deleted or null if it wasn't found
278     */
279    @Nullable
280    protected NodeMetadata doDestroyNode(final String id) {
281       checkNotNull(id, "id");
282       logger.debug(">> destroying node(%s)", id);
283       final AtomicReference<NodeMetadata> node = new AtomicReference<NodeMetadata>();
284       RetryablePredicate<String> tester = new RetryablePredicate<String>(new Predicate<String>() {
285 
286          @Override
287          public boolean apply(String input) {
288             try {
289                NodeMetadata md = destroyNodeStrategy.destroyNode(id);
290                if (md != null)
291                   node.set(md);
292                return true;
293             } catch (IllegalStateException e) {
294                logger.warn("<< illegal state destroying node(%s)", id);
295                return false;
296             }
297          }
298 
299       }, timeouts.nodeRunning, 1000, TimeUnit.MILLISECONDS);
300       
301       boolean successful = tester.apply(id) && (node.get() == null || nodeTerminated.apply(node));
302       if (successful)
303          credentialStore.remove("node#" + id);
304       logger.debug("<< destroyed node(%s) success(%s)", id, successful);
305       return node.get();
306    }

509    /**
510     * {@inheritDoc}
511     */
512    @Override
513    public Map<NodeMetadata, ExecResponse> runScriptOnNodesMatching(Predicate<NodeMetadata> filter, Statement runScript,
514          RunScriptOptions options) throws RunScriptOnNodesException {
515 
516       checkNotNull(filter, "filter");
517       checkNotNull(runScript, "runScript");
518       checkNotNull(options, "options");
519 
520       Map<NodeMetadata, ExecResponse> goodNodes = newLinkedHashMap();
521       Map<NodeMetadata, Exception> badNodes = newLinkedHashMap();
522       Map<NodeMetadata, Future<ExecResponse>> responses = newLinkedHashMap();
523       Map<?, Exception> exceptions = ImmutableMap.<Object, Exception> of();
524 
525       initAdminAccess.visit(runScript);
526 
527       Iterable<? extends RunScriptOnNode> scriptRunners = transformNodesIntoInitializedScriptRunners(
528             nodesMatchingFilterAndNotTerminatedExceptionIfNotFound(filter), runScript, options, badNodes);
529       if (Iterables.size(scriptRunners) > 0) {
530          for (RunScriptOnNode runner : scriptRunners) {
531             responses.put(runner.getNode(), executor.submit(new RunScriptOnNodeAndAddToGoodMapOrPutExceptionIntoBadMap(
532                   runner, goodNodes, badNodes)));
533          }
534          exceptions = awaitCompletion(responses, executor, null, logger, "runScriptOnNodesMatching(" + filter + ")");
535       }
536 
537       Function<NodeMetadata, NodeMetadata> fn = persistNodeCredentials.ifAdminAccess(runScript);
538       badNodes = Maps2.transformKeys(badNodes, fn);
539       goodNodes = Maps2.transformKeys(goodNodes, fn);
540 
541       if (exceptions.size() > 0 || badNodes.size() > 0) {
542          throw new RunScriptOnNodesException(runScript, options, goodNodes, exceptions, badNodes);
543       }
544       return goodNodes;
545    }

571    /**
572     * {@inheritDoc}
573     */
574    @Override
575    public ExecResponse runScriptOnNode(String id, Statement runScript, RunScriptOptions options) {
576       NodeMetadata node = this.getNodeMetadata(id);
577       if (node == null)
578          throw new NoSuchElementException(id);
579       if (node.getState() != NodeState.RUNNING)
580          throw new IllegalStateException("node " + id
581                + " needs to be running before executing a script on it. current state: " + node.getState());
582       initAdminAccess.visit(runScript);
583       node = updateNodeWithCredentialsIfPresent(node, options);
584       ExecResponse response = runScriptOnNodeFactory.create(node, runScript, options).init().call();
585       persistNodeCredentials.ifAdminAccess(runScript).apply(node);
586       return response;
587    }

592    @Override
593    public ListenableFuture<ExecResponse> submitScriptOnNode(String id, final Statement runScript,
594          RunScriptOptions options) {
595       NodeMetadata node = this.getNodeMetadata(id);
596       if (node == null)
597          throw new NoSuchElementException(id);
598       if (node.getState() != NodeState.RUNNING)
599          throw new IllegalStateException("node " + id
600                + " needs to be running before executing a script on it. current state: " + node.getState());
601       initAdminAccess.visit(runScript);
602       final NodeMetadata node1 = updateNodeWithCredentialsIfPresent(node, options);
603       ListenableFuture<ExecResponse> response = runScriptOnNodeFactory.submit(node1, runScript, options);
604       response.addListener(new Runnable() {
605 
606          @Override
607          public void run() {
608             persistNodeCredentials.ifAdminAccess(runScript).apply(node1);
609          }
610 
611       }, executor);
612       return response;
613    }

615    private Iterable<? extends RunScriptOnNode> transformNodesIntoInitializedScriptRunners(
616          Iterable<? extends NodeMetadata> nodes, Statement script, RunScriptOptions options,
617          Map<NodeMetadata, Exception> badNodes) {
618       return filter(

65     protected class AddNode implements Callable<AtomicReference<NodeMetadata>> {
66        private final String name;
67        private final String group;
68        private final Template template;
69  
70        public AddNode(String name, String group, Template template) {
71           this.name = checkNotNull(name, "name");
72           this.group = checkNotNull(group, "group");
73           this.template = checkNotNull(template, "template");
74        }
75  
76        @Override
77        public AtomicReference<NodeMetadata> call() throws Exception {
78           NodeMetadata node = null;
79           logger.debug(">> adding node location(%s) name(%s) image(%s) hardware(%s)", template.getLocation().getId(),
80                    name, template.getImage().getProviderId(), template.getHardware().getProviderId());
81           node = addNodeWithGroupStrategy.createNodeWithGroupEncodedIntoName(group, name, template);
82           logger.debug("<< %s node(%s)", node.getState(), node.getId());
83           return new AtomicReference<NodeMetadata>(node);
84        }
85  
86        public String toString() {
87           return toStringHelper(this).add("name", name).add("group", group).add("template", template).toString();
88        }

115    /**
116     * This implementation gets a list of acceptable node names to encode the group into, then it
117     * simultaneously runs the nodes and applies options to them.
118     */
119    @Override
120    public Map<?, Future<Void>> execute(String group, int count, Template template, Set<NodeMetadata> goodNodes,
121             Map<NodeMetadata, Exception> badNodes, Multimap<NodeMetadata, CustomizationResponse> customizationResponses) {
122       Map<String, Future<Void>> responses = newLinkedHashMap();
123       for (String name : getNextNames(group, template, count)) {
124          responses.put(name, compose(createNodeInGroupWithNameAndTemplate(group, name, template),
125                   customizeNodeAndAddToGoodMapOrPutExceptionIntoBadMapFactory.create(template.getOptions(), goodNodes,
126                            badNodes, customizationResponses), executor));
127       }
128       return responses;
129    }

131    /**
132     * This calls logic necessary to create a node and convert it from its provider-specific object
133     * to the jclouds {@link NodeMetadata} object. This call directly precedes customization, such as
134     * executing scripts.
135     * 
136     * </p> The outcome of this operation does not imply the node is {@link NodeState#RUNNING
137     * running}. If you want to insert logic after the node is created, yet before an attempt to
138     * customize the node, then append your behaviour to this method.
139     * 
140     * ex. to attach an ip address post-creation
141     * 
142     * <pre>
143     * &#064;Override
144     * protected Future&lt;AtomicReference&lt;NodeMetadata&gt;&gt; createNodeInGroupWithNameAndTemplate(String group, String name,
145     *          Template template) {
146     * 
147     *    Future&lt;AtomicReference&lt;NodeMetadata&gt;&gt; future = super.addNodeIntoGroupWithNameAndTemplate(group, name, template);
148     *    return Futures.compose(future, new Function&lt;AtomicReference&lt;NodeMetadata&gt;, AtomicReference&lt;NodeMetadata&gt;&gt;() {
149     * 
150     *       &#064;Override
151     *       public AtomicReference&lt;NodeMetadata&gt; apply(AtomicReference&lt;NodeMetadata&gt; input) {
152     *          NodeMetadata node = input.get();
153     *          // allocate and attach an ip
154     *          input.set(NodeMetadataBuilder.fromNodeMetadata(node).publicAddresses(ImmutableSet.of(ip.getIp())).build());
155     *          return input;
156     *       }
157     * 
158     *    }, executor);
159     * }
160     * </pre>
161     * 
162     * @param group group the node belongs to
163     * @param name generated name of the node
164     * @param template user-specified template
165     * @return node that is created, yet not necessarily in {@link NodeState#RUNNING}
166     */
167    protected Future<AtomicReference<NodeMetadata>> createNodeInGroupWithNameAndTemplate(String group, String name,
168             Template template) {
169       return executor.submit(new AddNode(name, group, template));
170    }

172    /**
173     * Find the next node names that can be used. These will be derived from the group and the
174     * template. We will pre-allocate a specified quantity, and attempt to verify that there is no
175     * name conflict with the current service.
176     * 
177     * @param group
178     * @param count
179     * @param template
180     * @return
181     */
182    protected Set<String> getNextNames(final String group, final Template template, int count) {
183       Set<String> names = newLinkedHashSet();
184       Iterable<? extends ComputeMetadata> currentNodes = listNodesStrategy.listNodes();
185       int maxTries = 100;
186       int currentTries = 0;
187       while (names.size() < count && currentTries++ < maxTries) {
188          final String name = namingConvention.createWithoutPrefix().uniqueNameForGroup(group);
189          if (!any(currentNodes, new Predicate<ComputeMetadata>() {
190 
191             @Override
192             public boolean apply(ComputeMetadata input) {
193                return name.equals(input.getName());
194             }
195 
196          })) {
197             names.add(name);
198          }
199       }
200       return names;
201    }

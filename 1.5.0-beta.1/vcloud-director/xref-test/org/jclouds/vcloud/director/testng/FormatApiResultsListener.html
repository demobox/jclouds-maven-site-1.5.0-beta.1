19  package org.jclouds.vcloud.director.testng;
20  
21  import java.text.SimpleDateFormat;
22  import java.util.Arrays;
23  import java.util.Set;
24  
25  import org.slf4j.Logger;
26  import org.slf4j.LoggerFactory;
27  import org.testng.ITestResult;
28  import org.testng.TestListenerAdapter;
29  
30  import com.google.common.base.Joiner;
31  import com.google.common.base.Optional;
32  import com.google.common.base.Predicates;
33  import com.google.common.base.Strings;
34  import com.google.common.collect.ImmutableSet;
35  import com.google.common.collect.Iterables;
36  
37  /**
38   * Outputs test status to the {@code jclouds.vcloud.api} logger.
39   * 
40   * Adapted from {@link org.jclouds.test.testng.UnitTestTestNGListener}.
41   * 
42   * @author Adrian Cole
43   */
44  public class FormatApiResultsListener extends TestListenerAdapter {
45  
46     public static final Logger logger = LoggerFactory.getLogger("jclouds.vcloud.api");
47  
48     private static final SimpleDateFormat timestamp = new SimpleDateFormat("yyyy-MM-dd HH:mm:ss");
49     private static final Set<String> apis = ImmutableSet.of("admin", "user");
50  
51     @Override
52     synchronized public void onTestSuccess(ITestResult res) {
53        if (methodInApiGroup(res)) {
54           String statusLine = resultForState(res, "succeeded");
55           logger.info(statusLine);
56        }
57     }
58  
59     @Override
60     synchronized public void onTestFailure(ITestResult res) {
61        if (methodInApiGroup(res)) {
62           String statusLine = resultForState(res, "failed");
63           logger.info(statusLine);
64        }
65     }
66  
67     @Override
68     synchronized public void onTestSkipped(ITestResult res) {
69        if (methodInApiGroup(res)) {
70           String statusLine = resultForState(res, "skipped");
71           logger.info(statusLine);
72        }
73     }
74  
75     private boolean methodInApiGroup(ITestResult res) {
76        return Iterables.any(Arrays.asList(res.getMethod().getGroups()), Predicates.in(apis));
77     }
78  
79     private String resultForState(ITestResult res, String state) {
80        return Joiner.on(',').join(getApi(res), getOperation(res), getStart(res), getTest(res), getDuration(res), state);
81     }
82  
83     private String getApi(ITestResult res) {
84        Optional<String> found = Iterables.tryFind(Arrays.asList(res.getMethod().getGroups()), Predicates.in(apis));
85        return found.isPresent() ? found.get() : "";
86     }
87  
88     private String getOperation(ITestResult res) {
89        return Strings.nullToEmpty(res.getMethod().getDescription());
90     }
91  
92     private String getTest(ITestResult res) {
93        return Strings.nullToEmpty(res.getName());
94     }
95  
96     private String getStart(ITestResult res) {
97        return timestamp.format(res.getStartMillis());
98     }
99  
100    private String getDuration(ITestResult res) {
101       return Long.toString(res.getEndMillis() - res.getStartMillis());
102    }
103 }

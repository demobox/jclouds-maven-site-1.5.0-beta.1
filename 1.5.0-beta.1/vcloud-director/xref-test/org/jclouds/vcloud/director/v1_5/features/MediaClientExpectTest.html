138                .xmlFilePayload("/media/error400.xml", VCloudDirectorMediaType.ERROR)
139                .httpResponseBuilder().statusCode(400).build());
140  
141       Error expected = Error.builder()
142             .message("validation error on field 'id': String value has invalid format or length")
143             .majorErrorCode(400)
144             .minorErrorCode("BAD_REQUEST")
145             .build();
146        
147       try {
148          client.getMediaClient().getMedia(mediaUri);
149          fail("Should give HTTP 400 error");
150       } catch (VCloudDirectorException vde) {
151          assertEquals(vde.getError(), expected);
152       } catch (Exception e) {
153          fail("Should have thrown a VCloudDirectorException");
154       }

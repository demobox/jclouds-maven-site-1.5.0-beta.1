52  @Test(groups = { "live", "user" }, singleThreaded = true, testName = "NetworkClientLiveTest")
53  public class NetworkClientLiveTest extends BaseVCloudDirectorClientLiveTest {
54     
55     public static final String NETWORK = "network";
56   
57     /*
58      * Convenience reference to API client.
59      */
60     protected NetworkClient networkClient;
61     
62     private boolean metadataSet = false;
63      
64     @Override
65     @BeforeClass(alwaysRun = true)
66     public void setupRequiredClients() {
67        networkClient = context.getApi().getNetworkClient();
68     }
69     
70     @AfterClass(alwaysRun = true)
71     public void cleanUp() {
72        if (metadataSet) {
73           try {
74  	         Task delete = adminContext.getApi().getNetworkClient().getMetadataClient().deleteMetadataEntry(toAdminUri(networkURI), "key");
75  	         taskDoneEventually(delete);
76           } catch (Exception e) {
77              logger.warn(e, "Error when deleting metadata");
78           }
79        }
80     }
81     
82     @Test(description = "GET /network/{id}")
83     public void testGetNetwork() {
84        // required for testing
85        assertNotNull(networkURI, String.format(REF_REQ_LIVE, NETWORK));
86         
87        Network abstractNetwork = networkClient.getNetwork(networkURI);
88        assertTrue(abstractNetwork instanceof OrgNetwork, String.format(REQUIRED_VALUE_OBJECT_FMT, 
89              ".class", NETWORK, abstractNetwork.getClass(),"OrgNetwork"));
90        OrgNetwork network = Network.toSubType(abstractNetwork);
91        assertNotNull(network, String.format(OBJ_REQ_LIVE, NETWORK));
92        assertTrue(!network.getDescription().equals("DO NOT USE"), "Network isn't to be used for testing");
93         
94        Checks.checkOrgNetwork(network);
95     }
96     
97     private void setupMetadata() {
98        adminContext.getApi().getNetworkClient().getMetadataClient().setMetadata(toAdminUri(networkURI), 
99              "key", MetadataValue.builder().value("value").build());
100       metadataSet = true;
101    }
102    
103    @Test(description = "GET /network/{id}/metadata", dependsOnMethods = { "testGetNetwork" })
104    public void testGetMetadata() {
105       if (adminContext != null) {
106          setupMetadata();
107       }
108       
109       Metadata metadata = networkClient.getMetadataClient().getMetadata(networkURI);
110       // required for testing
111       assertFalse(Iterables.isEmpty(metadata.getMetadataEntries()), 
112             String.format(OBJ_FIELD_REQ_LIVE, NETWORK, "metadata.entries"));
113        
114       // parent type
115       checkResourceType(metadata);
116        
117       for (MetadataEntry entry : metadata.getMetadataEntries()) {
118          // required elements and attributes
119          assertNotNull(entry.getKey(), 
120                String.format(OBJ_FIELD_ATTRB_REQ, networkClient, "MetadataEntry", entry.getKey(), "key"));
121          assertNotNull(entry.getValue(), 
122                String.format(OBJ_FIELD_ATTRB_REQ, networkClient, "MetadataEntry", entry.getValue(), "value"));
123           
124          // parent type
125          checkResourceType(entry);
126       }
127    }
128    
129    @Test(description = "GET /network/{id}/metadata/{key}", dependsOnMethods = { "testGetMetadata" })
130    public void testGetMetadataValue() {
131       MetadataValue metadataValue = networkClient.getMetadataClient().getMetadataValue(networkURI, "key");
132        
133       // Check parent type
134       checkResourceType(metadataValue);
135        
136       // Check required elements and attributes
137       String value = metadataValue.getValue();
138       assertNotNull(value, 
139             String.format(OBJ_FIELD_ATTRB_REQ, NETWORK, "MetadataEntry", 
140                   metadataValue.toString(), "value"));
141       assertEquals(value, "value", 
142             String.format(OBJ_FIELD_EQ, NETWORK, "metadataEntry.value", "value", value));
143    }

156    @Test(description = "POST /admin/user/{id}/action/unlock", dependsOnMethods = { "testUpdateUser" })
157    public void testUnlockUser() {
158       // Need to know how many times to fail login to lock account
159       AdminOrgClient adminOrgClient = adminContext.getApi().getOrgClient();
160       OrgPasswordPolicySettings settingsToRevertTo = null;
161 
162       // session client isn't typically exposed to the user, as it is implicit
163       SessionClient sessionClient = context.utils().injector().getInstance(SessionClient.class);
164       
165       OrgPasswordPolicySettings settings = adminOrgClient.getSettings(orgRef.getHref()).getPasswordPolicy();
166       assertNotNull(settings);
167 
168       // Adjust account settings so we can lock the account - be careful to not set invalidLoginsBeforeLockout too low!
169       if (!settings.isAccountLockoutEnabled()) {
170          settingsToRevertTo = settings;
171          settings = settings.toBuilder().accountLockoutEnabled(true).invalidLoginsBeforeLockout(5).build();
172          settings = adminOrgClient.updatePasswordPolicy(orgRef.getHref(), settings);
173       }
174 
175       assertTrue(settings.isAccountLockoutEnabled());
176       
177       for (int i = 0; i < settings.getInvalidLoginsBeforeLockout() + 1; i++) {
178          try {
179             sessionClient.loginUserInOrgWithPassword(URI.create(endpoint + "/sessions"), user.getName(), orgRef.getName(), "wrongpassword!");
180             fail("Managed to login using the wrong password!");
181          } catch (AuthorizationException e) {
182          } catch (Exception e) {
183             fail("Expected AuthorizationException", e);
184          }
185       }
186       
187       user = userClient.getUser(user.getHref());
188       assertTrue(user.isLocked());
189 
190       try {
191          sessionClient.loginUserInOrgWithPassword(URI.create(endpoint + "/sessions"), user.getName(), orgRef.getName(), "newPassword");
192          fail("Managed to login to locked account!");
193       } catch (AuthorizationException e) {
194       } catch (Exception e) {
195          fail("Expected AuthorizationException", e);
196       }
197       
198       userClient.unlockUser(user.getHref());
199 
200       user = userClient.getUser(user.getHref());
201       assertFalse(user.isLocked());
202 
203       // Double-check the user can now login again
204       SessionWithToken sessionWithToken = sessionClient.loginUserInOrgWithPassword(URI.create(endpoint + "/sessions"), user.getName(), orgRef.getName(), "newPassword");
205       assertNotNull(sessionWithToken.getToken());
206       sessionClient.logoutSessionWithToken(sessionWithToken.getSession().getHref(), sessionWithToken.getToken());
207       
208       // Return account settings to the previous values, if necessary
209       if (settingsToRevertTo != null) {
210          adminOrgClient.updatePasswordPolicy(orgRef.getHref(), settingsToRevertTo);
211       }
212    }

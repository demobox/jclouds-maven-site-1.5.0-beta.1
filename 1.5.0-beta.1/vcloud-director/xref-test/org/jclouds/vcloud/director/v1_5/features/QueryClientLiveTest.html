220    private static void assertRecordTypes(QueryResultRecords queryResult, Collection<String> validTypes, Class<?> validClazz) {
221       for (QueryResultRecordType record : queryResult.getRecords()) {
222          assertTrue(validTypes.contains(record.getType()), "invalid type for query result record, "+record.getType()+"; valid types are "+validTypes);
223          assertEquals(record.getClass(), validClazz, "invalid type for query result record, "+record.getClass()+"; expected "+validClazz);
224       }
225    }
226    
227    private Set<URI> toHrefs(QueryResultRecords queryResult) {
228       Set<URI> hrefs = new LinkedHashSet<URI>();
229       for (QueryResultRecordType record : queryResult.getRecords()) {
230          hrefs.add(record.getHref());
231       }
232       return hrefs;
233    }
234    
235    private Set<URI> toHrefs(Iterable<? extends Resource> resources) {
236       Set<URI> hrefs = new LinkedHashSet<URI>();
237       for (Resource resource : resources) {
238          hrefs.add(resource.getHref());
239       }
240       return hrefs;
241    }

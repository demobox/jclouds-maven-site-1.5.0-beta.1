67  @ConfiguresRestClient
68  public class NovaEC2RestClientModule extends EC2RestClientModule<NovaEC2Client, NovaEC2AsyncClient> {
69     public static final Map<Class<?>, Class<?>> DELEGATE_MAP = ImmutableMap.<Class<?>, Class<?>> builder()//
70           .put(AMIClient.class, AMIAsyncClient.class)//
71           .put(ElasticIPAddressClient.class, ElasticIPAddressAsyncClient.class)//
72           .put(InstanceClient.class, InstanceAsyncClient.class)//
73           .put(NovaEC2KeyPairClient.class, NovaEC2KeyPairAsyncClient.class)//
74           .put(SecurityGroupClient.class, SecurityGroupAsyncClient.class)//
75           .put(WindowsClient.class, WindowsAsyncClient.class)//
76           .put(AvailabilityZoneAndRegionClient.class, AvailabilityZoneAndRegionAsyncClient.class)//
77           .put(ElasticBlockStoreClient.class, ElasticBlockStoreAsyncClient.class)//
78           .build();
79  
80     public NovaEC2RestClientModule() {
81        super(TypeToken.of(NovaEC2Client.class), TypeToken.of(NovaEC2AsyncClient.class), DELEGATE_MAP);
82     }
83     
84     @Override
85     protected void configure() {
86        install(new NovaEC2ParserModule());
87        super.configure();
88        bind(CreateVolumeResponseHandler.class).to(NovaCreateVolumeResponseHandler.class).in(Scopes.SINGLETON);
89        bind(DescribeImagesResponseHandler.class).to(NovaDescribeImagesResponseHandler.class);
90     }
91  
92     @Override
93     protected void installLocations() {
94        install(new LocationModule());
95        bind(RegionIdToZoneIdsSupplier.class).to(DescribeAvailabilityZonesInRegion.class).in(Scopes.SINGLETON);
96        // there is only one region, and its endpoint is the same as the provider
97        bind(ZoneIdsSupplier.class).to(ZoneIdsFromRegionIdToZoneIdsValues.class).in(Scopes.SINGLETON);
98     }

101    @Provides
102    EC2Client provide(NovaEC2Client in) {
103       return in;
104    }

106    @Singleton
107    @Provides
108    EC2AsyncClient provide(NovaEC2AsyncClient in) {
109       return in;
110    }

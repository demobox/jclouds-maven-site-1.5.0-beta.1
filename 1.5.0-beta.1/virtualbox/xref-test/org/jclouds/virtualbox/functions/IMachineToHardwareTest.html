22  import static org.easymock.EasyMock.createNiceMock;
23  import static org.easymock.EasyMock.eq;
24  import static org.easymock.EasyMock.expect;
25  import static org.easymock.EasyMock.replay;
26  import static org.testng.Assert.assertEquals;
27  
28  import org.jclouds.compute.domain.Hardware;
29  import org.jclouds.compute.predicates.ImagePredicates;
30  import org.testng.annotations.Test;
31  import org.virtualbox_4_1.IGuestOSType;
32  import org.virtualbox_4_1.IMachine;
33  import org.virtualbox_4_1.IVirtualBox;
34  import org.virtualbox_4_1.VirtualBoxManager;
35  
36  import com.google.common.base.Suppliers;
37  
38  @Test(groups = "unit")
39  public class IMachineToHardwareTest {
40  
41     @Test
42     public void testConvert() throws Exception {
43        VirtualBoxManager vbm = createNiceMock(VirtualBoxManager.class);
44        IVirtualBox vBox = createNiceMock(IVirtualBox.class);
45        IMachine vm = createNiceMock(IMachine.class);
46        IGuestOSType guestOsType = createNiceMock(IGuestOSType.class);
47  
48        String linuxDescription = "Ubuntu Linux 10.04";
49        String machineName = "hw-machineId";
50  
51        expect(vm.getOSTypeId()).andReturn("os-type").anyTimes();
52        expect(vm.getName()).andReturn(machineName).anyTimes();
53  
54        expect(vm.getDescription()).andReturn(linuxDescription).anyTimes();
55  
56        expect(vBox.getGuestOSType(eq("os-type"))).andReturn(guestOsType);
57        expect(vbm.getVBox()).andReturn(vBox);
58        expect(guestOsType.getIs64Bit()).andReturn(true);
59  
60        replay(vbm, vBox, vm, guestOsType);
61  
62        Hardware hardware = new IMachineToHardware(Suppliers.ofInstance(vbm)).apply(vm);
63  
64        assertEquals(hardware.getId(), machineName);
65        assertEquals(hardware.getProviderId(), machineName);
66        // for starters assume 1-to-1 relationship hardware to image (which
67        // correlate to a single source IMachine)
68        assertEquals(hardware.supportsImage().toString(), ImagePredicates.idEquals(machineName).toString());
69  
70     }

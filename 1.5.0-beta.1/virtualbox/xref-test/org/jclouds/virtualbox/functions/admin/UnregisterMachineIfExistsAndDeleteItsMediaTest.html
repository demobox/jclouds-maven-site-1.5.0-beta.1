29  import java.util.List;

31  import org.jclouds.virtualbox.domain.HardDisk;

33  import org.jclouds.virtualbox.domain.VmSpec;
34  import org.testng.annotations.Test;
35  import org.virtualbox_4_1.CleanupMode;
36  import org.virtualbox_4_1.IMachine;
37  import org.virtualbox_4_1.IMedium;

39  import org.virtualbox_4_1.IVirtualBox;

43  @Test(groups = "unit", testName = "UnregisterMachineIfExistsTest")
44  public class UnregisterMachineIfExistsAndDeleteItsMediaTest {
45  
46     private String ideControllerName = "IDE Controller";
47     private CleanupMode mode = CleanupMode.Full;
48     private String vmName = "jclouds-image-example-machine-to-be-destroyed";
49     private String vmId = "jclouds-image-iso-unregister";
50     private String osTypeId = "";
51  
52     @Test
53     public void testUnregisterExistingMachine() throws Exception {
54        VirtualBoxManager manager = createMock(VirtualBoxManager.class);
55        IVirtualBox vBox = createMock(IVirtualBox.class);
56        IMachine registeredMachine = createMock(IMachine.class);
57        IProgress progress = createNiceMock(IProgress.class);
58        List<IMedium> media = new ArrayList<IMedium>();
59        List<IMedium> mediums = Collections.unmodifiableList(media);
60        
61        StorageController ideController = StorageController.builder().name(ideControllerName).bus(StorageBus.IDE)
62                .attachISO(0, 0, "/tmp/ubuntu-11.04-server-i386.iso")
63                .attachHardDisk(HardDisk.builder().diskpath("/tmp/testadmin.vdi").controllerPort(0).deviceSlot(1).build())
64                .attachISO(1, 1, "/tmp/VBoxGuestAdditions_4.1.2.iso").build();
65        VmSpec vmSpecification = VmSpec.builder().id(vmId).name(vmName).memoryMB(512).osTypeId(osTypeId)
66                .controller(ideController)
67                .forceOverwrite(true)
68                .cleanUpMode(CleanupMode.Full).build();
69  
70        expect(manager.getVBox()).andReturn(vBox).anyTimes();
71        expect(vBox.findMachine(vmName)).andReturn(registeredMachine);
72  
73        expect(registeredMachine.unregister(mode)).andReturn(mediums);
74        expectLastCall().anyTimes();
75        
76        expect(registeredMachine.delete(mediums)).andReturn(progress);
77        expectLastCall().anyTimes();
78  
79        replay(manager, vBox, registeredMachine, progress);
80  
81        new UnregisterMachineIfExistsAndDeleteItsMedia(vmSpecification).apply(registeredMachine);
82     }

20  package org.jclouds.virtualbox.statements;
21  
22  import static com.google.common.base.Preconditions.checkNotNull;
23  
24  import org.jclouds.scriptbuilder.domain.OsFamily;
25  import org.jclouds.scriptbuilder.domain.Statement;
26  import org.jclouds.virtualbox.domain.NetworkInterfaceCard;
27  
28  import com.google.common.collect.ImmutableList;
29  
30  /**
31   * Sets the ipaddress using ssh. Used for host-only networking
32   * 
33   * @author dralves
34   * 
35   */
36  public class SetIpAddress implements Statement {
37  
38     private String script;
39  
40     public SetIpAddress(NetworkInterfaceCard networkInterfaceCard) {
41        String ipAddress = networkInterfaceCard.getNetworkAdapter().getStaticIp();
42        checkNotNull(ipAddress, "ip address");
43        int slot = (int) networkInterfaceCard.getSlot();
44        String iface = null;
45        switch (slot) {
46           case 0:
47              iface = "eth0";
48              break;
49           case 1:
50              iface = "eth1";
51              break;
52           case 2:
53              iface = "eth2";
54              break;
55           case 3:
56              iface = "eth3";
57              break;
58           default:
59              throw new IllegalArgumentException("slot must be 0,1,2,3 (was: " + slot + ")");
60        }
61        script = String.format("ifconfig %s %s;", iface, ipAddress);
62     }
63  
64     @Override
65     public Iterable<String> functionDependencies(OsFamily family) {
66        return ImmutableList.of();
67     }
68  
69     @Override
70     public String render(OsFamily family) {
71        if (checkNotNull(family, "family") == OsFamily.WINDOWS)
72           throw new UnsupportedOperationException("windows not yet implemented");
73        return script;
74     }
75  
76  }

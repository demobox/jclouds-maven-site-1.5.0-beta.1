71  /**
72   * Creates nodes, by cloning a master vm and based on the provided {@link NodeSpec}. Must be
73   * synchronized mainly because of snapshot creation (must be synchronized on a per-master-basis).

75   * @author David Alves, Andrea Turli

102    @Override
103    public synchronized NodeAndInitialCredentials<IMachine> apply(NodeSpec nodeSpec) {
104 
105       checkNotNull(nodeSpec, "NodeSpec");
106 
107       Master master = nodeSpec.getMaster();
108       checkNotNull(master, "Master");
109 
110       if (master.getMachine().getCurrentSnapshot() != null) {
111          ISession session;
112          try {
113             session = manager.get().openMachineSession(master.getMachine());
114          } catch (Exception e) {
115             throw new RuntimeException("error opening vbox machine session: " + e.getMessage(), e);
116          }
117          IProgress progress = session.getConsole().deleteSnapshot(master.getMachine().getCurrentSnapshot().getId());
118          progress.waitForCompletion(-1);
119          session.unlockMachine();
120       }
121       String masterNameWithoutPrefix = master.getMachine().getName().replace(VIRTUALBOX_IMAGE_PREFIX, "");
122 
123       String cloneName = VIRTUALBOX_NODE_PREFIX + masterNameWithoutPrefix + VIRTUALBOX_NODE_NAME_SEPARATOR
124                + nodeSpec.getTag() + VIRTUALBOX_NODE_NAME_SEPARATOR + nodeSpec.getName();
125 
126       int ram = 512;
127       if (nodeSpec.getTemplate() != null && nodeSpec.getTemplate().getHardware() != null
128                && nodeSpec.getTemplate().getHardware().getRam() > 0) {
129          ram = nodeSpec.getTemplate().getHardware().getRam();
130       }
131       
132       VmSpec cloneVmSpec = VmSpec.builder().id(cloneName).name(cloneName).memoryMB(ram).cleanUpMode(CleanupMode.Full)
133                .forceOverwrite(true).build();
134 
135       // CASE NAT + HOST-ONLY
136       NetworkAdapter natAdapter = NetworkAdapter.builder().networkAttachmentType(NetworkAttachmentType.NAT)
137                .build();
138       NetworkInterfaceCard natIfaceCard = NetworkInterfaceCard.builder().addNetworkAdapter(natAdapter).slot(1L).build();
139 
140       NetworkAdapter hostOnlyAdapter = NetworkAdapter.builder().networkAttachmentType(NetworkAttachmentType.HostOnly)
141                .build();
142 
143       // create new hostOnly interface if needed, otherwise use the one already there with dhcp enabled ...
144       String hostOnlyIfName = getHostOnlyIfOrCreate();
145       
146       NetworkInterfaceCard hostOnlyIfaceCard = NetworkInterfaceCard.builder().addNetworkAdapter(hostOnlyAdapter)
147                .addHostInterfaceName(hostOnlyIfName).slot(0L).build();
148 
149       NetworkSpec networkSpec = createNetworkSpecForHostOnlyNATNICs(natIfaceCard, hostOnlyIfaceCard);
150 
151       CloneSpec cloneSpec = CloneSpec.builder().linked(true).master(master.getMachine()).network(networkSpec)
152                .vm(cloneVmSpec).build();
153 
154       IMachine cloned = cloner.apply(cloneSpec);
155 
156       machineController.ensureMachineIsLaunched(cloneVmSpec.getVmName());
157 
158       // IMachineToNodeMetadata produces the final ip's but these need to be set before so we build a
159       // NodeMetadata just for the sake of running the gshadow and setip scripts 
160       NodeMetadata partialNodeMetadata = buildPartialNodeMetadata(cloned);
161 
162       // see DeleteGShadowLock for a detailed explanation
163       machineUtils.runScriptOnNode(partialNodeMetadata, new DeleteGShadowLock(), RunScriptOptions.NONE);
164 
165       // CASE NAT + HOST-ONLY
166       machineUtils.runScriptOnNode(partialNodeMetadata, new EnableNetworkInterface(natIfaceCard), RunScriptOptions.NONE);
167       
168 
169       // TODO get credentials from somewhere else (they are also HC in
170       // IMachineToSshClient)
171       NodeAndInitialCredentials<IMachine> nodeAndInitialCredentials = new NodeAndInitialCredentials<IMachine>(cloned,
172                cloneName, LoginCredentials.builder().user("toor").password("password").authenticateSudo(true).build());
173 
174       return nodeAndInitialCredentials;
175    }

177    private NodeMetadata buildPartialNodeMetadata(IMachine clone) {
178       NodeMetadataBuilder nodeMetadataBuilder = new NodeMetadataBuilder();
179       nodeMetadataBuilder.id(clone.getName());
180       nodeMetadataBuilder.state(VirtualBoxComputeServiceContextModule.machineToNodeState.get(clone.getState()));
181       nodeMetadataBuilder.publicAddresses(ImmutableSet.of(machineUtils.getIpAddressFromHostOnlyNIC(clone.getName())));
182       
183       LoginCredentials loginCredentials = new LoginCredentials("toor", "password", null, true);
184       nodeMetadataBuilder.credentials(loginCredentials);
185       
186       return  nodeMetadataBuilder.build();
187    }

189    private NetworkSpec createNetworkSpecForHostOnlyNATNICs(NetworkInterfaceCard natIfaceCard,
190             NetworkInterfaceCard hostOnlyIfaceCard) {
191       return NetworkSpec.builder().addNIC(natIfaceCard).addNIC(hostOnlyIfaceCard).build();
192    }

197    private String getHostOnlyIfOrCreate() {     
198       
199       IHostNetworkInterface availableHostInterfaceIf = returnExistingHostNetworkInterfaceWithDHCPenabledOrNull(manager
200                .get().getVBox().getHost().getNetworkInterfaces());
201       if (availableHostInterfaceIf==null) {
202          final String hostOnlyIfName = createHostOnlyIf();
203          assignDHCPtoHostOnlyInterface(hostOnlyIfName);
204          return hostOnlyIfName;
205       } else {
206          return availableHostInterfaceIf.getName();
207       }

211    private void assignDHCPtoHostOnlyInterface(final String hostOnlyIfName) {
212       List<IHostNetworkInterface> availableNetworkInterfaces = manager.get().getVBox().getHost()
213                .getNetworkInterfaces();
214       
215       IHostNetworkInterface iHostNetworkInterfaceWithHostOnlyIfName = Iterables.getOnlyElement(Iterables.filter(availableNetworkInterfaces, new Predicate<IHostNetworkInterface>() {
216 
217          @Override
218          public boolean apply(IHostNetworkInterface iHostNetworkInterface) {
219             return iHostNetworkInterface.getName().equals(hostOnlyIfName);
220          }
221       }));
222       
223       String hostOnlyIfIpAddress = iHostNetworkInterfaceWithHostOnlyIfName.getIPAddress();
224       String dhcpIpAddress = hostOnlyIfIpAddress.substring(0, hostOnlyIfIpAddress.lastIndexOf(".")) + ".254";
225       String dhcpNetmask = "255.255.255.0";
226       String dhcpLowerIp = hostOnlyIfIpAddress.substring(0, hostOnlyIfIpAddress.lastIndexOf(".")) + ".2";
227       String dhcpUpperIp = hostOnlyIfIpAddress.substring(0, hostOnlyIfIpAddress.lastIndexOf(".")) + ".253";
228       
229       ExecResponse response = runScriptOnNodeFactory
230                .create(host.get(),
231                         Statements.exec(String
232                                  .format("VBoxManage dhcpserver add --ifname %s --ip %s --netmask %s --lowerip %s --upperip %s --enable",
233                                           hostOnlyIfName, dhcpIpAddress, dhcpNetmask, dhcpLowerIp, dhcpUpperIp)), runAsRoot(false).wrapInInitScript(false)).init().call();
234       checkState(response.getExitStatus()==0);
235       /*
236       runScriptOnNodeFactory
237                .create(host.get(),
238                         Statements.exec(String.format("VBoxManage hostonlyif ipconfig %s --ip %s",
239                                  hostOnlyIfName, hostOnlyIfIpAddress)), runAsRoot(false).wrapInInitScript(false)).init().call();
240       */
241    }

243    private String createHostOnlyIf() {
244       final String hostOnlyIfName;
245       ExecResponse createHostOnyResponse = runScriptOnNodeFactory
246                .create(host.get(), Statements.exec("VBoxManage hostonlyif create"),
247                         runAsRoot(false).wrapInInitScript(false)).init().call();
248       String output = createHostOnyResponse.getOutput();
249       checkState(createHostOnyResponse.getExitStatus()==0);
250       checkState(output.contains("'"), "cannot create hostonlyif");
251       hostOnlyIfName = output.substring(output.indexOf("'") + 1, output.lastIndexOf("'"));
252       return hostOnlyIfName;
253    }

255    private IHostNetworkInterface returnExistingHostNetworkInterfaceWithDHCPenabledOrNull(Iterable<IHostNetworkInterface> availableNetworkInterfaces) {
256       checkNotNull(availableNetworkInterfaces);
257       return Iterables.getFirst(filterAvailableNetworkInterfaceByHostOnlyAndDHCPenabled(availableNetworkInterfaces), null);
258    }

260    /**
261     * @param availableNetworkInterfaces 
262     * @param hostOnlyIfIpAddress
263     * @return
264     */
265    private Iterable<IHostNetworkInterface> filterAvailableNetworkInterfaceByHostOnlyAndDHCPenabled(Iterable<IHostNetworkInterface> availableNetworkInterfaces) {
266       Iterable<IHostNetworkInterface> filteredNetworkInterfaces = Iterables.filter(availableNetworkInterfaces, new Predicate<IHostNetworkInterface>() {
267          @Override
268          public boolean apply(IHostNetworkInterface iHostNetworkInterface) {
269             // this is an horrible workaround cause iHostNetworkInterface.getDhcpEnabled is working only for windows host
270             boolean match = false;
271             List<IDHCPServer> availableDHCPservers = manager.get().getVBox().getDHCPServers();
272             for (IDHCPServer idhcpServer : availableDHCPservers) {
273                if(idhcpServer.getEnabled() && idhcpServer.getNetworkName().equals(iHostNetworkInterface.getNetworkName()))
274                   match  = true;
275             }
276             return iHostNetworkInterface.getInterfaceType().equals(HostNetworkInterfaceType.HostOnly) &&
277                     match;
278             }
279       });
280       return filteredNetworkInterfaces;
281    }

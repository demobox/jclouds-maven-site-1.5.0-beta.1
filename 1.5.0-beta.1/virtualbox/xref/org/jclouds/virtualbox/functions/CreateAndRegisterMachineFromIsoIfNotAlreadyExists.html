60  @Singleton
61  public class CreateAndRegisterMachineFromIsoIfNotAlreadyExists implements Function<MasterSpec, IMachine> {
62  
63     @Resource
64     @Named(ComputeServiceConstants.COMPUTE_LOGGER)
65     protected Logger logger = Logger.NULL;
66  
67     private final Supplier<VirtualBoxManager> manager;
68     private final MachineUtils machineUtils;
69  
70     private final String workingDir;
71  
72     @Inject
73     public CreateAndRegisterMachineFromIsoIfNotAlreadyExists(Supplier<VirtualBoxManager> manager,
74              MachineUtils machineUtils, @Named(VirtualBoxConstants.VIRTUALBOX_WORKINGDIR) String workingDir) {
75        this.manager = manager;
76        this.machineUtils = machineUtils;
77        this.workingDir = workingDir;
78     }
79  
80     @Override
81     public IMachine apply(@Nullable MasterSpec launchSpecification) {
82        final IVirtualBox vBox = manager.get().getVBox();
83        String vmName = launchSpecification.getVmSpec().getVmName();
84        String vmId = launchSpecification.getVmSpec().getVmId();
85  
86        try {
87           vBox.findMachine(vmId);
88           throw new IllegalStateException("Machine " + vmName + " is already registered.");
89        } catch (VBoxException e) {
90           if (machineNotFoundException(e))
91              return createMachine(vBox, launchSpecification);
92           else
93              throw e;
94        }
95     }
96  
97     private IMachine createMachine(IVirtualBox vBox, MasterSpec machineSpec) {
98        VmSpec vmSpec = machineSpec.getVmSpec();
99        String settingsFile = vBox.composeMachineFilename(vmSpec.getVmName(), workingDir);
100 
101       IMachine newMachine = vBox.createMachine(settingsFile, vmSpec.getVmName(), vmSpec.getOsTypeId(),
102                vmSpec.getVmId(), vmSpec.isForceOverwrite());
103       manager.get().getVBox().registerMachine(newMachine);
104       ensureConfiguration(machineSpec);
105       return newMachine;
106    }
107 
108    private void ensureConfiguration(MasterSpec machineSpec) {
109       VmSpec vmSpec = machineSpec.getVmSpec();
110       NetworkSpec networkSpec = machineSpec.getNetworkSpec();
111       String vmName = vmSpec.getVmName();
112 
113       // Change RAM
114       ensureMachineHasMemory(vmName, vmSpec.getMemory());
115 
116       Set<StorageController> controllers = vmSpec.getControllers();
117       if (controllers.isEmpty()) {
118          throw new IllegalStateException(missingIDEControllersMessage(vmSpec));
119       }
120       StorageController controller = controllers.iterator().next();
121       ensureMachineHasStorageControllerNamed(vmName, controller);
122       setupHardDisksForController(vmName, controller);
123       setupDvdsForController(vmSpec, vmName, controller);
124 
125       // Networking
126       for (NetworkInterfaceCard networkInterfaceCard : networkSpec.getNetworkInterfaceCards()) {
127          new AttachNicToMachine(vmName, machineUtils).apply(networkInterfaceCard);
128       }
129    }
130 
131    private void setupDvdsForController(VmSpec vmSpecification, String vmName, StorageController controller) {
132       Set<IsoImage> dvds = controller.getIsoImages();
133       for (IsoImage dvd : dvds) {
134          String dvdSource = dvd.getSourcePath();
135          final IMedium dvdMedium = manager.get().getVBox()
136                   .openMedium(dvdSource, DeviceType.DVD, AccessMode.ReadOnly, vmSpecification.isForceOverwrite());
137          ensureMachineDevicesAttached(vmName, dvdMedium, dvd.getDeviceDetails(), controller.getName());
138       }
139    }
140 
141    private void ensureMachineDevicesAttached(String vmName, IMedium medium, DeviceDetails deviceDetails,
142             String controllerName) {
143       machineUtils.writeLockMachineAndApply(vmName, new AttachMediumToMachineIfNotAlreadyAttached(deviceDetails,
144                medium, controllerName));
145    }
146 
147    private String missingIDEControllersMessage(VmSpec vmSpecification) {
148       return String
149                .format("First controller is not an IDE controller. Please verify that the VM spec is a correct master node: %s",
150                         vmSpecification);
151    }
152 
153    private void setupHardDisksForController(String vmName, StorageController controller) {
154       Set<HardDisk> hardDisks = controller.getHardDisks();
155       for (HardDisk hardDisk : hardDisks) {
156          String sourcePath = hardDisk.getDiskPath();
157          if (new File(sourcePath).exists()) {
158             boolean deleted = new File(sourcePath).delete();
159             if (!deleted) {
160                logger.error(String.format("File %s could not be deleted.", sourcePath));
161             }
162          }
163          IMedium medium = new CreateMediumIfNotAlreadyExists(manager, machineUtils, true).apply(hardDisk);
164          ensureMachineDevicesAttached(vmName, medium, hardDisk.getDeviceDetails(), controller.getName());
165       }
166    }
167 
168    private void ensureMachineHasMemory(String vmName, final long memorySize) {
169       machineUtils.writeLockMachineAndApply(vmName, new ApplyMemoryToMachine(memorySize));
170    }
171 
172    public void ensureMachineHasStorageControllerNamed(String vmName, StorageController storageController) {
173       machineUtils.writeLockMachineAndApply(vmName,
174                new AddIDEControllerIfNotExists(checkNotNull(storageController, "storageController")));
175    }

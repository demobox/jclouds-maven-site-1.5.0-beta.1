86  @SuppressWarnings("unchecked")
87  public class VirtualBoxComputeServiceContextModule extends
88           ComputeServiceAdapterContextModule<IMachine, Hardware, Image, Location> {
89  
90     @Override
91     protected void configure() {
92        super.configure();
93        bind(new TypeLiteral<ComputeServiceAdapter<IMachine, Hardware, Image, Location>>() {
94        }).to(VirtualBoxComputeServiceAdapter.class);
95        bind(new TypeLiteral<Function<IMachine, NodeMetadata>>() {
96        }).to(IMachineToNodeMetadata.class);
97        bind(new TypeLiteral<Function<Location, Location>>() {
98        }).to((Class) IdentityFunction.class);
99        bind(new TypeLiteral<Function<Hardware, Hardware>>() {
100       }).to((Class) IdentityFunction.class);
101       bind(new TypeLiteral<Function<Image, Image>>() {
102       }).to((Class) IdentityFunction.class);
103       bind(new TypeLiteral<Function<IMachine, Hardware>>() {
104       }).to(IMachineToHardware.class);
105       bind(new TypeLiteral<Function<IMachine, Image>>() {
106       }).to(IMachineToImage.class);
107       bind(new TypeLiteral<CacheLoader<IsoSpec, URI>>() {
108       }).to((Class) PreseedCfgServer.class);
109       bind(new TypeLiteral<Function<URI, File>>() {
110       }).to((Class) FileDownloadFromURI.class);
111 
112       bind(new TypeLiteral<Supplier<VirtualBoxManager>>() {
113       }).to((Class) StartVBoxIfNotAlreadyRunning.class);
114       // the yaml config to image mapper
115       bind(new TypeLiteral<Supplier<Map<Image, YamlImage>>>() {
116       }).to((Class) ImagesToYamlImagesFromYamlDescriptor.class);
117       // the yaml config provider
118       bind(YamlImagesFromFileConfig.class);
119 
120       // the master machines cache
121       bind(new TypeLiteral<LoadingCache<Image, Master>>() {
122       }).to(MastersLoadingCache.class);
123 
124       // the vbox image extension
125       bind(new TypeLiteral<ImageExtension>() {
126       }).to(VirtualBoxImageExtension.class);
127 
128       // the master creating function
129       bind(new TypeLiteral<Function<MasterSpec, IMachine>>() {
130       }).to((Class) CreateAndInstallVm.class);
131       // the machine cloning function
132       bind(new TypeLiteral<Function<NodeSpec, NodeAndInitialCredentials<IMachine>>>() {
133       }).to((Class) NodeCreator.class);
134       bind(new TypeLiteral<Function<CloneSpec, IMachine>>() {
135       }).to((Class) CloneAndRegisterMachineFromIMachineIfNotAlreadyExists.class);
136       // the jetty server provider
137       bind(new TypeLiteral<Server>() {
138       }).to((Class) PreseedCfgServer.class).asEagerSingleton();
139 
140       bind(new TypeLiteral<Function<IMachine, SshClient>>() {
141       }).to(IMachineToSshClient.class);
142 
143       bind(ExecutionType.class).toInstance(ExecutionType.HEADLESS);
144       bind(LockType.class).toInstance(LockType.Write);
145    }

147    @Provides
148    @Singleton
149    protected Function<Supplier<NodeMetadata>, VirtualBoxManager> provideVBox() {
150       return new Function<Supplier<NodeMetadata>, VirtualBoxManager>() {
151 
152          @Override
153          public VirtualBoxManager apply(Supplier<NodeMetadata> nodeSupplier) {
154             return VirtualBoxManager.createInstance(nodeSupplier.get().getId());
155          }
156 
157          @Override
158          public String toString() {
159             return "createInstanceByNodeId()";
160          }
161 
162       };
163    }
164 
165    @Provides
166    @Singleton
167    protected Predicate<SshClient> sshResponds(SshResponds sshResponds, Timeouts timeouts) {
168       return new RetryablePredicate<SshClient>(sshResponds, timeouts.nodeRunning, 500l, TimeUnit.MILLISECONDS);
169    }
170 
171    @Override
172    protected Optional<ImageExtension> provideImageExtension(Injector i) {
173       return Optional.of(i.getInstance(ImageExtension.class));
174    }

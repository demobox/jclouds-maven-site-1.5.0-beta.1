331    public S orderedRequestsSendResponses(final List<HttpRequest> requests, final List<HttpResponse> responses) {
332       final AtomicInteger counter = new AtomicInteger(0);
333 
334       return createClient(new Function<HttpRequest, HttpResponse>() {
335          @Override
336          public HttpResponse apply(HttpRequest input) {
337             int index = counter.getAndIncrement();
338             if (index >= requests.size())
339                return HttpResponse.builder().statusCode(500).message(
340                         String.format("request %s is out of range (%s)", index, requests.size())).payload(
341                         Payloads.newStringPayload(renderRequest(input))).build();
342             if (!httpRequestsAreEqual(input, requests.get(index))) {
343                assertEquals(renderRequest(input), renderRequest(requests.get(index)));
344             }
345             return responses.get(index);
346          }
347       });

373    /**
374     * Compare two requests as instructed by {@link #compareHttpRequestAsType(HttpRequest)} - default
375     * is to compare using Objects.equal
376     */
377    public boolean httpRequestsAreEqual(HttpRequest a, HttpRequest b) {
378       try {
379          if (a == null || b == null || !Objects.equal(a.getRequestLine(), b.getRequestLine())
380                || !Objects.equal(a.getHeaders(), b.getHeaders())) {
381             return false;
382          }
383          if (a.getPayload() == null || b.getPayload() == null) {
384             return Objects.equal(a, b);
385          }
386  
387          switch (compareHttpRequestAsType(a)) {
388             case XML: {
389                Diff diff = XMLUnit.compareXML(Strings2.toStringAndClose(a.getPayload().getInput()), Strings2
390                         .toStringAndClose(b.getPayload().getInput()));
391 
392                // Ignoring whitespace in elements that have other children, xsi:schemaLocation and
393                // differences in namespace prefixes
394                diff.overrideDifferenceListener(new DifferenceListener() {
395                   @Override
396                   public int differenceFound(Difference diff) {
397                      if (diff.getId() == DifferenceConstants.SCHEMA_LOCATION_ID
398                               || diff.getId() == DifferenceConstants.NAMESPACE_PREFIX_ID) {
399                         return RETURN_IGNORE_DIFFERENCE_NODES_IDENTICAL;
400                      }
401                      if (diff.getId() == DifferenceConstants.TEXT_VALUE_ID) {
402                         for (NodeDetail detail : ImmutableSet.of(diff.getControlNodeDetail(), diff.getTestNodeDetail())) {
403                            if (detail.getNode().getParentNode().getChildNodes().getLength() < 2
404                                     || !detail.getValue().trim().isEmpty()) {
405                               return RETURN_ACCEPT_DIFFERENCE;
406                            }
407                         }
408                         return RETURN_IGNORE_DIFFERENCE_NODES_IDENTICAL;
409                      }
410                      return RETURN_ACCEPT_DIFFERENCE;
411                   }
412 
413                   @Override
414                   public void skippedComparison(Node node, Node node1) {
415                   }
416                });
417 
418                return diff.identical();
419             }
420             case JSON: {               
421                JsonParser parser = new JsonParser();
422                JsonElement payloadA = parser.parse(Strings2.toStringAndClose(a.getPayload().getInput()));
423                JsonElement payloadB = parser.parse(Strings2.toStringAndClose(b.getPayload().getInput()));
424                return Objects.equal(payloadA, payloadB);
425             }
426             default: {
427                return Objects.equal(a, b);
428             }
429          }
430       } catch (Exception e) {
431          throw Throwables.propagate(e);
432       }
433    }

439    public S requestsSendResponses(final Map<HttpRequest, HttpResponse> requestToResponse, Module module,
440             Properties props) {
441       return createClient(new Function<HttpRequest, HttpResponse>() {
442          ImmutableBiMap<HttpRequest, HttpResponse> bimap = ImmutableBiMap.copyOf(requestToResponse);
443 
444          @Override
445          public HttpResponse apply(HttpRequest input) {
446             HttpResponse response = null;
447             for (HttpRequest request : requestToResponse.keySet()) {
448                if (httpRequestsAreEqual(input, request)) {
449                   response = requestToResponse.get(request);
450                }
451             }
452 
453             if (response == null) {
454                StringBuilder payload = new StringBuilder("\n");
455                payload.append("the following request is not configured:\n");
456                payload.append("----------------------------------------\n");
457                payload.append(renderRequest(input));
458                payload.append("----------------------------------------\n");
459                payload.append("configured requests:\n");
460                for (HttpRequest request : requestToResponse.keySet()) {
461                   payload.append("----------------------------------------\n");
462                   payload.append(renderRequest(request));
463                }
464                response = HttpResponse.builder().statusCode(500).message("no response configured for request").payload(
465                         Payloads.newStringPayload(payload.toString())).build();
466 
467             } else if (compareHttpRequestAsType(input) == HttpRequestComparisonType.DEFAULT) {
468                // in case hashCode/equals doesn't do a full content check
469                assertEquals(renderRequest(input), renderRequest(bimap.inverse().get(response)));
470             }
471 
472             return response;
473          }
474       }, module, props);
475    }

477    public String renderRequest(HttpRequest request) {
478       StringBuilder builder = new StringBuilder().append(request.getRequestLine()).append('\n');
479       for (Entry<String, String> header : request.getHeaders().entries()) {
480          builder.append(header.getKey()).append(": ").append(header.getValue()).append('\n');
481       }
482       if (request.getPayload() != null) {
483          for (Entry<String, String> header : contentMetadataCodec.toHeaders(
484                   request.getPayload().getContentMetadata()).entries()) {
485             builder.append(header.getKey()).append(": ").append(header.getValue()).append('\n');
486          }
487          try {
488             builder.append('\n').append(Strings2.toStringAndClose(request.getPayload().getInput()));
489          } catch (IOException e) {
490             throw Throwables.propagate(e);
491          }
492 
493       } else {
494          builder.append('\n');
495       }
496       return builder.toString();
497    }

535    protected Injector createInjector(Function<HttpRequest, HttpResponse> fn, Module module, Properties props) {
536       ContextBuilder builder = null;
537       if (provider != null)
538          try {
539             builder = ContextBuilder.newBuilder(provider).credentials(identity, credential);
540          } catch (NoSuchElementException e) {
541             Logger
542                      .getAnonymousLogger()
543                      .warning(
544                               "provider ["
545                                        + provider
546                                        + "] is not setup as META-INF/services/org.jclouds.apis.ApiMetadata or META-INF/services/org.jclouds.providers.ProviderMetadata");
547          }
548       if (builder == null) {
549          ProviderMetadata pm = createProviderMetadata();
550          ApiMetadata am = (pm != null) ? pm.getApiMetadata() : checkNotNull(createApiMetadata(),
551                   "either createApiMetadata or createProviderMetadata must be overridden");
552 
553          builder = pm != null ? ContextBuilder.newBuilder(pm) : ContextBuilder.newBuilder(RestApiMetadata.class.cast(am));
554       }
555 
556       this.api = RestApiMetadata.class.cast(builder.getApiMetadata()).getApi();
557 
558       // isolate tests from eachother, as default credentialStore is static
559       return builder.credentials(identity, credential).modules(
560                ImmutableSet.of(new ExpectModule(fn), new NullLoggingModule(), new CredentialStoreModule(new CopyInputStreamInputSupplierMap(
561                      new ConcurrentHashMap<String, InputSupplier<InputStream>>())), module)).overrides(setupProperties())
562                .buildInjector();
563    }

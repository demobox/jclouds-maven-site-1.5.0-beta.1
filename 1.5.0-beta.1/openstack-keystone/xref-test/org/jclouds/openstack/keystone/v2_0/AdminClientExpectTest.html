61     public void testGetApiMetaData() {
62        AdminClient client = requestsSendResponses(
63              keystoneAuthWithUsernameAndPassword, responseWithKeystoneAccess,
64              standardRequestBuilder(endpoint + "/v2.0/").
65                    headers(ImmutableMultimap.of("Accept", APPLICATION_JSON)).build(),
66              standardResponseBuilder(200).payload(payloadFromResourceWithContentType("/api_metadata.json", APPLICATION_JSON)).build())
67              .getAdminClientForRegion("region-a.geo-1");
68        ApiMetadata metadata = client.getApiMetadata();
69        assertNotNull(metadata);
70        assertEquals(metadata.getId(), "v2.0");
71  
72        ApiMetadata expected = ApiMetadata.builder().id("v2.0")
73              .links(ImmutableSet.of(Link.builder().relation(Link.Relation.SELF).href(URI.create("http://172.16.89.140:5000/v2.0/")).build(),
74                    Link.builder().relation(Link.Relation.DESCRIBEDBY).type("text/html").href(URI.create("http://docs.openstack.org/api/openstack-identity-service/2.0/content/")).build(),
75                    Link.builder().relation(Link.Relation.DESCRIBEDBY).type("application/pdf").href(URI.create("http://docs.openstack.org/api/openstack-identity-service/2.0/identity-dev-guide-2.0.pdf")).build()
76              ))
77              .status("beta")
78              .updated(new SimpleDateFormatDateService().iso8601SecondsDateParse("2011-11-19T00:00:00Z"))
79              .mediaTypes(ImmutableSet.of(
80                    MediaType.builder().base("application/json").type("application/vnd.openstack.identity-v2.0+json").build(),
81                    MediaType.builder().base("application/xml").type("application/vnd.openstack.identity-v2.0+xml").build()
82              ))
83              .build();
84  
85        assertEquals(metadata, expected);
86     }

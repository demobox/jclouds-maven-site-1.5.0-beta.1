20  public class ParseLoginResponseFromHttpResponse implements Function<HttpResponse, LoginResponse> {

23  
24     @Inject
25     ParseLoginResponseFromHttpResponse(GsonWrapper gson) {
26        this.parser = new ParseFirstJsonValueNamed<LoginResponse>(checkNotNull(gson, "gsonView"),
27           new TypeLiteral<LoginResponse>(){}, "loginresponse");
28     }
29  
30     @Override
31     public LoginResponse apply(HttpResponse response) {
32        checkNotNull(response, "response");
33  
34        LoginResponse login =  parser.apply(response);
35        checkNotNull(login, "loginResponse");
36  
37        String jSessionId = get(Splitter.on("=").split(get(Splitter.on(";").trimResults().split(
38           getOnlyElement(response.getHeaders().get("Set-Cookie"))), 0)), 1);
39        
40        return LoginResponse.builder().copyOf(login).jSessionId(jSessionId).build();
41     }

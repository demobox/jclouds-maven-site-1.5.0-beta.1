51  
52        SSHKeyPairClient client = requestSendsResponse(HttpRequest.builder()
53           .method("GET")

56                 "command=listSSHKeyPairs&listAll=true&apiKey=identity&signature=5d2J9u%2BdKpkQsadDbl9i9OcUSLQ%3D"))

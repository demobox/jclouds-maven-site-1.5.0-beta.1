274    @Test(enabled = false, dependsOnMethods = "testCreateAndAttachVolume")
275    void testBundleInstance() {
276       SshClient ssh = sshFactory.create(HostAndPort.fromParts(instance.getIpAddress(), 22),
277             LoginCredentials.builder().user("ubuntu").privateKey(keyPair.getKeyMaterial()).build());
278       try {
279          ssh.connect();
280       } catch (SshException e) {// try twice in case there is a network timeout
281          try {
282             Thread.sleep(10 * 1000);
283          } catch (InterruptedException e1) {
284          }
285          ssh.connect();
286       }
287       try {
288          System.out.printf("%d: %s writing ebs script%n", System.currentTimeMillis(), instance.getId());
289          String script = "/tmp/mkebsboot-init.sh";
290          ssh.put(script, Payloads.newStringPayload(mkEbsBoot));
291 
292          System.out.printf("%d: %s launching ebs script%n", System.currentTimeMillis(), instance.getId());
293          ssh.exec("chmod 755 " + script);
294          ssh.exec(script + " init");
295          ExecResponse output = ssh.exec("sudo " + script + " start");
296          System.out.println(output);
297          output = ssh.exec(script + " status");
298 
299          assert !output.getOutput().trim().equals("") : output;
300 
301          RetryablePredicate<String> scriptTester = new RetryablePredicate<String>(new ScriptTester(ssh, SCRIPT_END),
302                600, 10, TimeUnit.SECONDS);
303          scriptTester.apply(script);
304       } finally {
305          if (ssh != null)
306             ssh.disconnect();
307       }
308    }

310    public static class ScriptTester implements Predicate<String> {
311       private final SshClient ssh;
312       private final String endMatches;
313 
314       public ScriptTester(SshClient ssh, String endMatches) {
315          this.ssh = ssh;
316          this.endMatches = endMatches;
317       }
318 
319       @Override
320       public boolean apply(String script) {
321          System.out.printf("%d: %s testing status%n", System.currentTimeMillis(), script);
322          ExecResponse output = ssh.exec(script + " status");
323          if (output.getOutput().trim().equals("")) {
324             output = ssh.exec(script + " tail");
325             String stdout = output.getOutput().trim();
326             if (stdout.contains(endMatches)) {
327                return true;
328             } else {
329                output = ssh.exec(script + " tailerr");
330                String stderr = output.getOutput().trim();
331                throw new RuntimeException(String.format(
332                      "script %s ended without token: stdout.log: [%s]; stderr.log: [%s]; ", script, stdout, stderr));
333             }
334          }
335          return false;
336       }
337 
338    }

341    void testAMIFromBundle() {
342       volume = Iterables.getOnlyElement(client.getElasticBlockStoreServices().describeVolumesInRegion(
343             volume.getRegion(), volume.getId()));
344       if (volume.getAttachments().size() > 0) {
345          // should be cleanly unmounted, so force is not necessary.
346          client.getElasticBlockStoreServices().detachVolumeInRegion(instance.getRegion(), volume.getId(), false);
347          System.out.printf("%d: %s awaiting detachment to complete%n", System.currentTimeMillis(), volume.getId());
348          assert volumeTester.apply(volume);
349       } else {
350          attachment = null; // protect test closer so that it doesn't try to
351          // detach
352       }
353       snapshot = client.getElasticBlockStoreServices().createSnapshotInRegion(volume.getRegion(), volume.getId(),
354             withDescription("EBS Ubuntu Hardy"));
355 
356       System.out.printf("%d: %s awaiting snapshot to complete%n", System.currentTimeMillis(), snapshot.getId());
357 
358       assert snapshotTester.apply(snapshot);
359       Image image = Iterables.getOnlyElement(client.getAMIServices().describeImagesInRegion(snapshot.getRegion(),
360             imageIds(IMAGE_ID)));
361       String description = image.getDescription() == null ? "jclouds" : image.getDescription();
362 
363       System.out.printf("%d: %s creating ami from snapshot%n", System.currentTimeMillis(), snapshot.getId());
364 
365       String amiId = client.getAMIServices().registerUnixImageBackedByEbsInRegion(
366             snapshot.getRegion(),
367             "ebsboot-" + image.getId(),
368             snapshot.getId(),
369             withKernelId(image.getKernelId()).withRamdisk(image.getRamdiskId()).withDescription(description)
370                   .asArchitecture(Architecture.I386));
371       try {
372          ebsImage = Iterables.getOnlyElement(client.getAMIServices().describeImagesInRegion(snapshot.getRegion(),
373                imageIds(amiId)));
374       } catch (AWSResponseException e) {
375          // TODO add a retry handler for this HTTP code 400 and the below error
376          if (e.getError().getClass().equals("InvalidAMIID.NotFound"))
377             ebsImage = Iterables.getOnlyElement(client.getAMIServices().describeImagesInRegion(snapshot.getRegion(),
378                   imageIds(amiId)));
379          else
380             throw e;
381       }
382       verifyImage();
383    }

526       System.out.printf("%d: %s awaiting instance to run %n", System.currentTimeMillis(), instance.getId());
527       assert runningTester.apply(instance);
528 
529       // search my identity for the instance I just created

531             .describeInstancesInRegion(instance.getRegion(), instance.getId()); // last
532       // parameter
533       // (ids)
534       // narrows
535       // the
536       // search
537 
538       instance = Iterables.getOnlyElement(Iterables.getOnlyElement(reservations));
539 
540       System.out.printf("%d: %s awaiting ssh service to start%n", System.currentTimeMillis(), instance.getIpAddress());
541       assert socketTester.apply(HostAndPort.fromParts(instance.getIpAddress(), 22));

543       sshPing(instance);
544       System.out.printf("%d: %s ssh connection made%n", System.currentTimeMillis(), instance.getId());
545       return instance;

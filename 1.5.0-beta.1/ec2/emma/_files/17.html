211   protected void cleanUpIncidentalResources(final String region, final String group){
212      // For issue #445, tries to delete security groups first: ec2 throws exception if in use, but
213      // deleting a key pair does not.
214      // This is "belt-and-braces" because deleteKeyPair also does extractIdsFromInstances & usingKeyPairAndNotDead
215      // for us to check if any instances are using the key-pair before we delete it. 
216      // There is (probably?) still a race if someone is creating instances at the same time as deleting them: 
217      // we may delete the key-pair just when the node-being-created was about to rely on the incidental 
218      // resources existing.
219 
220      // Also in #445, in aws-ec2 the deleteSecurityGroup sometimes fails after terminating the final VM using a 
221      // given security group, if called very soon after the VM's state reports terminated. Emprically, it seems that
222      // waiting a small time (e.g. enabling logging or debugging!) then the tests pass. We therefore retry.
223      final int maxAttempts = 3;
224      Retryables.retryNumTimes(new Predicate<Void>() {
225            @Override
226            public boolean apply(Void input) {
227               try {
228                  logger.debug(">> deleting incidentalResources(%s @ %s)", region, group);
229                  deleteSecurityGroup(region, group);
230                  deleteKeyPair(region, group); // not executed if securityGroup was in use
231                  logger.debug("<< deleted incidentalResources(%s @ %s)", region, group);
232                  return true;
233               } catch (IllegalStateException e) {
234                  logger.debug("<< inUse incidentalResources(%s @ %s)", region, group);
235                  return false;
236               }
237            }
238         }, (Void)null, maxAttempts);
239   }

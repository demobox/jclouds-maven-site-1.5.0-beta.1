105    @Override
106    public NodeAndInitialCredentials<ServerDetails> createNodeWithGroupEncodedIntoName(String group, String name,
107          Template template) {
108       checkNotNull(template, "template was null");
109       checkNotNull(template.getOptions(), "template options was null");
110       checkArgument(template.getOptions().getClass().isAssignableFrom(GleSYSTemplateOptions.class),
111             "options class %s should have been assignable from GleSYSTemplateOptions", template.getOptions().getClass());
112 
113       GleSYSTemplateOptions templateOptions = template.getOptions().as(GleSYSTemplateOptions.class);
114 
115       CreateServerOptions createServerOptions = new CreateServerOptions();
116       createServerOptions.ip(templateOptions.getIp());
117 
118       createServerOptions.description(name); // TODO: add to templateOptions and
119                                              // set if present
120 
121       ServerSpec.Builder builder = ServerSpec.builder();
122       builder.datacenter(template.getLocation().getId());
123       builder.templateName(template.getImage().getId());
124       builder.platform(template.getHardware().getHypervisor());
125       builder.memorySizeMB(template.getHardware().getRam());
126       builder.diskSizeGB(Math.round(template.getHardware().getVolumes().get(0).getSize()));
127       builder.cpuCores((int) template.getHardware().getProcessors().get(0).getCores());
128       builder.transferGB(50);// TODO: add to template options with default value
129       ServerSpec spec = builder.build();
130 
131       String password = passwordProvider.get(); // TODO: add to templateOptions
132                                                 // and set if present
133 
134       logger.debug(">> creating new Server spec(%s) name(%s) options(%s)", spec, name, createServerOptions);
135       ServerDetails result = client.getServerClient().createServerWithHostnameAndRootPassword(spec, name, password,
136             createServerOptions);
137       logger.trace("<< server(%s)", result.getId());
138 
139       return new NodeAndInitialCredentials<ServerDetails>(result, result.getId() + "", LoginCredentials.builder()
140             .password(password).build());
141    }

157    @Override
158    public Iterable<Hardware> listHardwareProfiles() {
159       Set<? extends Location> locationsSet = locations.get();
160       ImmutableSet.Builder<Hardware> hardwareToReturn = ImmutableSet.builder();
161 
162       // do this loop after dupes are filtered, else OOM
163       Set<OSTemplate> images = listImages();
164 
165       for (Entry<String, AllowedArgumentsForCreateServer> platformToArgs : client.getServerClient()
166             .getAllowedArgumentsForCreateServerByPlatform().entrySet())
167          for (String datacenter : platformToArgs.getValue().getDataCenters())
168             for (int diskSizeGB : platformToArgs.getValue().getDiskSizesInGB())
169                for (int cpuCores : platformToArgs.getValue().getCpuCoreOptions())
170                   for (int memorySizeMB : platformToArgs.getValue().getMemorySizesInMB()) {
171                      ImmutableSet.Builder<String> templatesSupportedBuilder = ImmutableSet.builder();
172                      for (OSTemplate template : images) {
173                         if (template.getPlatform().equals(platformToArgs.getKey())
174                               && diskSizeGB >= template.getMinDiskSize() && memorySizeMB >= template.getMinMemSize())
175                            templatesSupportedBuilder.add(template.getName());
176                      }
177                      ImmutableSet<String> templatesSupported = templatesSupportedBuilder.build();
178                      if (templatesSupported.size() > 0)
179                         hardwareToReturn.add(new HardwareBuilder()
180                               .ids(String.format(
181                                     "datacenter(%s)platform(%s)cpuCores(%d)memorySizeMB(%d)diskSizeGB(%d)", datacenter,
182                                     platformToArgs.getKey(), cpuCores, memorySizeMB, diskSizeGB)).ram(memorySizeMB)
183                               .processors(ImmutableList.of(new Processor(cpuCores, 1.0)))
184                               .volumes(ImmutableList.<Volume> of(new VolumeImpl((float) diskSizeGB, true, true)))
185                               .hypervisor(platformToArgs.getKey())
186                               .location(Iterables.find(locationsSet, LocationPredicates.idEquals(datacenter)))
187                               .supportsImage(ImagePredicates.idIn(templatesSupported)).build());
188                   }
189 
190       return hardwareToReturn.build();
191    }

228    @Override
229    public void destroyNode(String id) {
230       new RetryablePredicate<String>(new Predicate<String>() {
231 
232          @Override
233          public boolean apply(String arg0) {
234             try {
235                client.getServerClient().destroyServer(arg0, DestroyServerOptions.Builder.discardIp());
236                return true;
237             } catch (IllegalStateException e) {
238                return false;
239             }
240          }
241 
242       }, timeouts.nodeTerminated).apply(id);
243    }

28  import org.jclouds.glesys.options.AddRecordOptions;

32  /**
33   * Provides synchronous access to Domain requests.
34   * <p/>
35   *
36   * @author Adam Lowe
37   * @see DomainAsyncClient
38   * @see <a href="https://customer.glesys.com/api.php" />
39   */
40  @Timeout(duration = 30, timeUnit = TimeUnit.SECONDS)
41  public interface DomainClient {
42  
43     /**
44      * Get a list of all invoices for this account.
45      *
46      * @return an account's associated invoice objects.
47      */
48     Set<Domain> listDomains();
49  
50     /**
51      * Add a domain to the Glesys dns-system
52      *
53      * @param domain  the name of the domain to add.
54      * @param options optional parameters
55      */
56     void addDomain(String domain, AddDomainOptions... options);
57  
58     /**
59      * Add a domain to the Glesys dns-system
60      *
61      * @param domain  the name of the domain to add.
62      * @param options optional parameters
63      */
64     void editDomain(String domain, DomainOptions... options);
65  
66     /**
67      * Remove a domain to the Glesys dns-system
68      *
69      * @param domain the name of the domain to remove
70      */
71     void deleteDomain(String domain);
72  
73     /**
74      * Retrieve the DNS records for a given domain
75      *
76      * @param domain the name of the domain to retrieve records for
77      */
78     Set<DomainRecord> listRecords(String domain);
79  
80     /**
81      * Add a DNS Record
82      *
83      * @param domain the domain to add the record to
84      * @param host
85      * @param type
86      * @param data
87      * @param options optional settings for the record
88      */
89     void addRecord(String domain, String host, String type, String data, AddRecordOptions... options);
90  
91     /**
92      * Modify a specific DNS Record
93      *
94      * @param recordId the id for the record to edit
95      * @param options  the settings to change
96      * @see #listRecords to retrieve the necessary ids
97      */
98     void editRecord(String recordId, EditRecordOptions... options);
99  
100    /**
101     * Delete a DNS record
102     *
103     * @param recordId the id for the record to delete
104     * @see #listRecords to retrieve the necessary ids
105     */
106    void deleteRecord(String recordId);
107 
108 }

19     public static String getPlatform(ComputeMetadata jcloudsObject) {
20        checkNotNull(jcloudsObject, "jcloudsObject");
21        Matcher matcher = JCLOUDS_ID_TO_PLATFORM.matcher(jcloudsObject.getId());
22        if (!matcher.matches()) {
23           throw new IllegalArgumentException(jcloudsObject.getId() + " not a GleSYS platform-based id!");
24        }

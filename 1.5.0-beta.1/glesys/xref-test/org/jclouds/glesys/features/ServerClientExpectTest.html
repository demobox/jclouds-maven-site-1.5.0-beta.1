133                         .put("Accept", "application/json")
134                         .put("Authorization", "Basic aWRlbnRpdHk6Y3JlZGVudGlhbA==").build()).build(),

136 
137       ImmutableSet.Builder<OSTemplate> expectedBuilder = ImmutableSet.builder();
138 
139       for (String name : new String[] { "Centos 5", "Centos 5 64-bit", "Centos 6 32-bit", "Centos 6 64-bit",
140             "Debian 5.0 32-bit", "Debian 5.0 64-bit", "Debian 6.0 32-bit", "Debian 6.0 64-bit", "Fedora Core 11",
141             "Fedora Core 11 64-bit", "Gentoo", "Gentoo 64-bit", "Scientific Linux 6", "Scientific Linux 6 64-bit",

143          expectedBuilder.add(new OSTemplate(name, 5, 128, "linux", "OpenVZ"));
144       }
145 
146       for (String name : new String[] { "CentOS 5.5 x64", "CentOS 5.5 x86", "Centos 6 x64", "Centos 6 x86",
147             "Debian-6 x64", "Debian 5.0.1 x64", "FreeBSD 8.2", "Gentoo 10.1 x64", "Ubuntu 8.04 x64",
148             "Ubuntu 10.04 LTS 64-bit", "Ubuntu 10.10 x64", "Ubuntu 11.04 x64" }) {
149          expectedBuilder.add(new OSTemplate(name, 5, 512, name.startsWith("FreeBSD") ? "freebsd" : "linux", "Xen"));
150       }
151       for (String name : new String[] { "Windows Server 2008 R2 x64 std", "Windows Server 2008 R2 x64 web",
152             "Windows Server 2008 x64 web", "Windows Server 2008 x86 web" }) {
153          expectedBuilder.add(new OSTemplate(name, 20, 1024, "windows", "Xen"));
154       }
155       
156       assertEquals(client.listTemplates(), expectedBuilder.build());
157    }

33  /**
34   * Tests behavior of {@code IpClient}
35   *
36   * @author Adrian Cole, Mattias Holmqvist

38  @Test(groups = "live", testName = "IpClientLiveTest")
39  public class IpClientLiveTest extends BaseGleSYSClientLiveTest {
40  
41     @BeforeGroups(groups = {"live"})
42     public void setupContext() {
43        super.setupContext();
44        client = gleContext.getApi().getIpClient();
45     }
46  
47     private IpClient client;
48  
49     @Test
50     public void testListFree() throws Exception {
51        Set<String> freeIps = client.listFree("4", "Falkenberg", "Xen");
52        assertTrue(freeIps.size() >= 1);
53     }
54  
55     @Test
56     public void testGetOpenVZDetails() throws Exception {
57        Set<String> openVzIps = client.listFree("4", "Falkenberg", "OpenVZ");
58        assertTrue(openVzIps.size() >= 1);
59        String openVzIp = openVzIps.iterator().next();
60        IpDetails ipDetails = client.getIpDetails(openVzIp);
61        assertEquals(ipDetails.getDatacenter(), "Falkenberg");
62        assertEquals(ipDetails.getPlatform(), "OpenVZ");
63        assertEquals(ipDetails.getIpversion(), "4");
64        
65        // TODO: Ask Glesys to include address in response for OpenVZ?
66        // assertEquals(ipDetails.getHostText(), openVzIp);
67     }
68  
69     @Test
70     public void testGetXenDetails() throws Exception {
71        Set<String> xenVzIps = client.listFree("4", "Falkenberg", "Xen");
72        assertTrue(xenVzIps.size() >= 1);
73        String xenIp = xenVzIps.iterator().next();
74        IpDetails ipDetails = client.getIpDetails(xenIp);
75        assertEquals(ipDetails.getDatacenter(), "Falkenberg");
76        assertEquals(ipDetails.getPlatform(), "Xen");
77        assertEquals(ipDetails.getIpversion(), "4");
78        assertEquals(ipDetails.getAddress(), xenIp);
79        assertNotNull(ipDetails.getPtr());
80        assertNotNull(ipDetails.getBroadcast());
81        assertNotNull(ipDetails.getGateway());
82        assertNotNull(ipDetails.getNetmask());
83        List<String> nameServers = ipDetails.getNameServers();
84        assertNotNull(nameServers);
85     }

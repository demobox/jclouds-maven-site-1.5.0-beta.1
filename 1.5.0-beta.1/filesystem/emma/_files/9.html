496    * {@inheritDoc}
497    */
498   @Override
499   public ListenableFuture<Blob> getBlob(final String containerName, final String key, GetOptions options) {
500      logger.debug("Retrieving blob with key %s from container %s", key, containerName);
501      // If the container doesn't exist, an exception is thrown
502      if (!storageStrategy.containerExists(containerName)) {
503         logger.debug("Container %s does not exist", containerName);
504         return immediateFailedFuture(cnfe(containerName));
505      }
506      // If the blob doesn't exist, a null object is returned
507      if (!storageStrategy.blobExists(containerName, key)) {
508         logger.debug("Item %s does not exist in container %s", key, containerName);
509         return immediateFuture(null);
510      }
511 
512      Blob blob = loadBlob(containerName, key);
513 
514      if (options != null) {
515         if (options.getIfMatch() != null) {
516            if (!blob.getMetadata().getETag().equals(options.getIfMatch()))
517               return immediateFailedFuture(returnResponseException(412));
518         }
519         if (options.getIfNoneMatch() != null) {
520            if (blob.getMetadata().getETag().equals(options.getIfNoneMatch()))
521               return immediateFailedFuture(returnResponseException(304));
522         }
523         if (options.getIfModifiedSince() != null) {
524            Date modifiedSince = options.getIfModifiedSince();
525            if (blob.getMetadata().getLastModified().before(modifiedSince)) {
526               HttpResponse response = new HttpResponse(304, null, null);
527               return immediateFailedFuture(new HttpResponseException(String.format("%1$s is before %2$s", blob
528                     .getMetadata().getLastModified(), modifiedSince), null, response));
529            }
530 
531         }
532         if (options.getIfUnmodifiedSince() != null) {
533            Date unmodifiedSince = options.getIfUnmodifiedSince();
534            if (blob.getMetadata().getLastModified().after(unmodifiedSince)) {
535               HttpResponse response = new HttpResponse(412, null, null);
536               return immediateFailedFuture(new HttpResponseException(String.format("%1$s is after %2$s", blob
537                     .getMetadata().getLastModified(), unmodifiedSince), null, response));
538            }
539         }
540         blob = copyBlob(blob);
541 
542         if (options.getRanges() != null && options.getRanges().size() > 0) {
543            byte[] data;
544            try {
545               data = toByteArray(blob.getPayload().getInput());
546            } catch (IOException e) {
547               return immediateFailedFuture(new RuntimeException(e));
548            }
549            ByteArrayOutputStream out = new ByteArrayOutputStream();
550            for (String s : options.getRanges()) {
551               // HTTP uses a closed interval while Java array indexing uses a
552               // half-open interval.
553               int offset = 0;
554               int last = data.length - 1;
555               if (s.startsWith("-")) {
556                  offset = last - Integer.parseInt(s.substring(1)) + 1;
557               } else if (s.endsWith("-")) {
558                  offset = Integer.parseInt(s.substring(0, s.length() - 1));
559               } else if (s.contains("-")) {
560                  String[] firstLast = s.split("\\-");
561                  offset = Integer.parseInt(firstLast[0]);
562                  last = Integer.parseInt(firstLast[1]);
563               } else {
564                  return immediateFailedFuture(new IllegalArgumentException("illegal range: " + s));
565               }
566 
567               if (offset > last) {
568                  return immediateFailedFuture(new IllegalArgumentException("illegal range: " + s));
569               }
570               if (last + 1 > data.length) {
571                  last = data.length - 1;
572               }
573               out.write(data, offset, last - offset + 1);
574            }
575            ContentMetadata cmd = blob.getPayload().getContentMetadata();
576            byte[] byteArray = out.toByteArray();
577            blob.setPayload(byteArray);
578            HttpUtils.copy(cmd, blob.getPayload().getContentMetadata());
579            blob.getPayload().getContentMetadata().setContentLength(new Long(byteArray.length));
580         }
581      }
582      checkNotNull(blob.getPayload(), "payload " + blob);
583      return immediateFuture(blob);
584   }

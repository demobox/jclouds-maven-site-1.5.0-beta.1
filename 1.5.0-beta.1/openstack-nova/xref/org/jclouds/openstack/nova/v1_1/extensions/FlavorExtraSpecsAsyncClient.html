52  /**
53   * Provide access to extra metadata for Nova flavors.
54   *
55   * @author Adam Lowe
56   * @see <a href="http://nova.openstack.org/api/nova.api.openstack.compute.contrib.flavorextraspecs.html"/>
57   * @see org.jclouds.openstack.nova.v1_1.features.FlavorClient
58   * @see FlavorExtraSpecsClient
59   */
60  @Extension(of = ServiceType.COMPUTE, namespace = ExtensionNamespaces.FLAVOR_EXTRA_SPECS)
61  @Timeout(duration = 180, timeUnit = TimeUnit.SECONDS)
62  @RequestFilters(AuthenticateRequest.class)
63  @Consumes(MediaType.APPLICATION_JSON)
64  public interface FlavorExtraSpecsAsyncClient {
65  
66     /**
67      * @see FlavorExtraSpecsClient#getAllExtraSpecs(String)
68      */
69     @GET
70     @SelectJson("extra_specs")
71     @Path("/flavors/{flavor_id}/os-extra_specs")
72     @ExceptionParser(ReturnEmptyMapOnNotFoundOr404.class)
73     ListenableFuture<Map<String, String>> getAllExtraSpecs(@PathParam("flavor_id") String flavorId);
74  
75     /**
76      * @see FlavorExtraSpecsClient#setExtraSpec(String, String, String)
77      */
78     @POST
79     @Path("/flavors/{flavor_id}/os-extra_specs")
80     @Produces(MediaType.APPLICATION_JSON)
81     @ExceptionParser(ReturnFalseOnNotFoundOr404.class)
82     @MapBinder(BindToJsonPayload.class)
83     ListenableFuture<Boolean> setAllExtraSpecs(@PathParam("flavor_id") String flavorId, @PayloadParam("extra_specs") Map<String, String> specs);
84  
85     /**
86      * @see FlavorExtraSpecsClient#getExtraSpec(String, String)
87      */
88     @GET
89     @Path("/flavors/{flavor_id}/os-extra_specs/{key}")
90     @Unwrap
91     @ExceptionParser(ReturnNullOnNotFoundOr404.class)
92     ListenableFuture<String> getExtraSpec(@PathParam("flavor_id") String flavorId, @PathParam("key") String key);
93  
94     /**
95      * @see FlavorExtraSpecsClient#setExtraSpec(String, String, String)
96      */
97     @PUT
98     @Path("/flavors/{flavor_id}/os-extra_specs/{key}")
99     @Produces(MediaType.APPLICATION_JSON)
100    @ExceptionParser(ReturnFalseOnNotFoundOr404.class)
101    @Payload("%7B\"{key}\":\"{value}\"%7D")
102    ListenableFuture<Boolean> setExtraSpec(@PathParam("flavor_id") String flavorId,
103                                           @PathParam("key") @PayloadParam("key") String key,
104                                           @PayloadParam("value") String value);
105 
106    /**
107     * @see FlavorExtraSpecsClient#deleteExtraSpec(String, String)
108     */
109    @DELETE
110    @Path("/flavors/{flavor_id}/os-extra_specs/{key}")
111    @ExceptionParser(ReturnFalseOnNotFoundOr404.class)
112    ListenableFuture<Boolean> deleteExtraSpec(@PathParam("flavor_id") String flavorId,
113                                              @PathParam("key") String key);

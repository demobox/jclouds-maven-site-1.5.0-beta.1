54     /**
55      * Servers contain a status attribute that can be used as an indication of the current server
56      * state. Servers with an ACTIVE status are available for use.
57      * 
58      * Other possible values for the status attribute include: BUILD, REBUILD, SUSPENDED, RESIZE,
59      * VERIFY_RESIZE, REVERT_RESIZE, PASSWORD, REBOOT, HARD_REBOOT, DELETED, UNKNOWN, and ERROR.
60      * 
61      * @author Adrian Cole
62      */
63     public static enum Status {
64  
65        ACTIVE(NodeState.RUNNING), BUILD(NodeState.PENDING), REBUILD(NodeState.PENDING), SUSPENDED(NodeState.SUSPENDED), RESIZE(
66                 NodeState.PENDING), VERIFY_RESIZE(NodeState.PENDING), REVERT_RESIZE(NodeState.PENDING), PASSWORD(
67                 NodeState.PENDING), REBOOT(NodeState.PENDING), HARD_REBOOT(NodeState.PENDING), DELETED(
68                 NodeState.TERMINATED), UNKNOWN(NodeState.UNRECOGNIZED), ERROR(NodeState.ERROR), UNRECOGNIZED(
69                 NodeState.UNRECOGNIZED), PAUSED(NodeState.SUSPENDED);
70  
71        protected final NodeState nodeState;
72  
73        Status(NodeState nodeState) {
74           this.nodeState = nodeState;
75        }
76  
77        public String value() {
78           return name();
79        }
80  
81        public static Status fromValue(String v) {
82           try {
83              return valueOf(v.replaceAll("\\(.*", ""));
84           } catch (IllegalArgumentException e) {
85              return UNRECOGNIZED;
86           }
87        }
88  
89        public NodeState getNodeState() {
90           return nodeState;
91        }
92     }

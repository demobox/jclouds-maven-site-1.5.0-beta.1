73     public static abstract class FieldValueResponseParser<T> implements Function<HttpResponse, Boolean> {
74        private final T expectedValue;
75        private final ParseFirstJsonValueNamed<T> valueParser;
76  
77        public FieldValueResponseParser(GsonWrapper wrapper, String fieldName, T expectedValue) {
78           valueParser = new ParseFirstJsonValueNamed<T>(wrapper, new TypeLiteral<T>() {
79           }, fieldName);
80           this.expectedValue = expectedValue;
81        }
82  
83        @Override
84        public Boolean apply(HttpResponse response) {
85           return Objects.equal(expectedValue, valueParser.apply(response));
86        }
